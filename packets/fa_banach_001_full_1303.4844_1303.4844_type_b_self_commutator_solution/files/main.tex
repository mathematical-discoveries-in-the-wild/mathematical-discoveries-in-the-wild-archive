\begin{abstract}
Let $J$ be a conjugation on a separable infinite-dimensional complex Hilbert
space $H$, and let
\[
 \ooJ=\{X\in\K(H):X=-JX^*J\}
\]
be the type-(B) compact orthogonal Lie algebra.  We prove that every
self-adjoint $T\in\ooJ$ is a self-commutator $T=[Y^*,Y]$ of an element
$Y\in\ooJ$.  In fact $Y$ can be chosen with
$\norm{Y}\leq\sqrt2\norm{T}^{1/2}$.  Consequently the possible eigenvalue
multisets are exactly the real null multisets in which every nonzero
$\lambda$ and $-\lambda$ have equal multiplicity.  The compactness issue is
resolved by grouping two spectral pairs at a time into four-dimensional
orthogonal blocks; an explicit square-root estimate makes their direct sum
compact.
\end{abstract}

\section{Notation and elementary consequences}

Throughout, $H$ is a separable infinite-dimensional complex Hilbert space,
$J$ is a conjugation, and
\[
 \ooJ:=\{X\in\K(H):X=-JX^*J\}.
\]
For matrix units, write $L_{jk}=E_{jk}-E_{kj}$.  Thus $L_{jk}$ is real
skew-symmetric and $L_{jk}^*=-L_{jk}$.

If $T=T^*\in\ooJ$, then
\[
 T=-JTJ,\qquad TJ=-JT.                                      \tag{2.1}
\]
Hence $J$ maps the $\lambda$-eigenspace antiunitarily onto the
$(-\lambda)$-eigenspace.  In particular, nonzero eigenvalues occur in
opposite pairs with equal multiplicity.

On a finite-dimensional $J$-invariant subspace, choose a $J$-real
orthonormal basis.  In that basis $J$ is coordinatewise conjugation and the
relation $X=-JX^*J$ becomes
\[
 X^T=-X.                                                     \tag{2.2}
\]
Thus the relevant finite blocks are ordinary complex skew-symmetric matrices.

\section{Proof intuition}

A single spectral pair $\{\lambda,-\lambda\}$ lives in a two-dimensional
$J$-invariant space, but $\mathfrak{so}(2,\C)$ is abelian, so it cannot
produce a nonzero self-commutator.  The obstruction disappears as soon as two
pairs are combined: an explicit $\mathfrak{so}(4,\C)$ formula produces both
pairs at once.  If a finite-rank operator leaves one pair over, its
infinite-dimensional kernel supplies a $J$-fixed zero vector and an
$\mathfrak{so}(3,\C)$ formula handles the resulting block.

The important quantitative feature is not merely finite-dimensional
solvability.  The formulas give
$\norm{Y_k}\leq\sqrt2\norm{T_k}^{1/2}$ on every block.  Since the eigenvalues
of compact $T$ tend to zero, the norms of the block solutions tend to zero.
Their orthogonal direct sum is therefore compact and remains in the same
type-(B) algebra.

\section{Finite block formulas}

\begin{lemma}[three- and four-dimensional blocks]\label{lem:blocks}
Let $m\in\{3,4\}$ and let $S\in M_m(\C)$ satisfy $S=S^*$ and $S^T=-S$.
Then there is $Z\in M_m(\C)$ such that
\[
 Z^T=-Z,\qquad [Z^*,Z]=S,\qquad
 \norm{Z}\leq\sqrt2\norm{S}^{1/2}.                         \tag{4.1}
\]
\end{lemma}

\begin{proof}
Since $S=S^*$ and $S^T=-S$, we have $\overline S=-S$, so $S=iR$ for a
real skew-symmetric matrix $R$.  The real orthogonal normal form for $R$
shows that, after conjugating by a real orthogonal matrix, the target is
\[
 i\lambda L_{12}\quad(m=3),
 \qquad
 i\lambda L_{12}+i\mu L_{34}\quad(m=4),                     \tag{4.2}
\]
where $\lambda\geq\mu\geq0$.  Real orthogonal conjugacy preserves transpose,
adjoint, commutators, and operator norm, so it suffices to solve these forms.

For $m=3$, use $[L_{13},L_{23}]=-L_{12}$ and set
\[
 Z=\sqrt{\frac{\lambda}{2}}\,(L_{13}+iL_{23}).              \tag{4.3}
\]
Then $Z^T=-Z$ and
\[
 [Z^*,Z]
 =-2i\frac{\lambda}{2}[L_{13},L_{23}]
 =i\lambda L_{12}.
\]
Because $\norm{L_{jk}}=1$, the triangle inequality gives
$\norm Z\leq\sqrt{2\lambda}$.

For $m=4$, put
\[
\begin{array}{lll}
 A_1=(L_{12}+L_{34})/2,&A_2=(L_{13}-L_{24})/2,
   &A_3=(L_{14}+L_{23})/2,\\
 B_1=(L_{12}-L_{34})/2,&B_2=(L_{13}+L_{24})/2,
   &B_3=(L_{14}-L_{23})/2.
\end{array}                                                   \tag{4.4}
\]
Direct multiplication gives
\[
 [A_2,A_3]=-A_1,\qquad [B_2,B_3]=B_1,
 \qquad [A_j,B_k]=0.                                        \tag{4.5}
\]
Define
\[
 Z=\sqrt{\frac{\lambda+\mu}{2}}(A_2+iA_3)
   +\sqrt{\frac{\lambda-\mu}{2}}(B_2-iB_3).                \tag{4.6}
\]
Every summand is skew-symmetric.  Equations (4.5), including the vanishing
of all cross terms, yield
\begin{align*}
 [Z^*,Z]
 &=i(\lambda+\mu)A_1+i(\lambda-\mu)B_1\\
 &=i\lambda L_{12}+i\mu L_{34}.
\end{align*}
The operators $A_j$ and $B_j$ have norm $1/2$, so
$\norm{A_2+iA_3}\leq1$ and $\norm{B_2-iB_3}\leq1$.  Therefore
\begin{align*}
 \norm Z
 &\leq \sqrt{\frac{\lambda+\mu}{2}}
       +\sqrt{\frac{\lambda-\mu}{2}}\\
 &\leq \sqrt{2\lambda}
  =\sqrt2\norm S^{1/2}.
\end{align*}
This proves the lemma.
\end{proof}

\section{Main theorem}

\begin{theorem}[type-(B) self-commutator theorem]\label{thm:main}
For every self-adjoint $T\in\ooJ$ there exists $Y\in\ooJ$ such that
\[
 [Y^*,Y]=T
 \quad\text{and}\quad
 \norm Y\leq\sqrt2\norm T^{1/2}.                            \tag{5.1}
\]
\end{theorem}

\begin{proof}
The assertion is immediate for $T=0$, so assume $T\neq0$.  By the compact
self-adjoint spectral theorem and (2.1), choose an orthonormal list
$(u_n)$ of eigenvectors for the positive eigenvalues, repeated with
multiplicity, so that
\[
 Tu_n=\lambda_nu_n,\qquad T(Ju_n)=-\lambda_nJu_n,
 \qquad \lambda_n>0.                                       \tag{5.2}
\]
The two vectors in each pair are orthogonal because they belong to distinct
eigenspaces.  If the list is infinite, arrange it so that
$\lambda_n\downarrow0$.

For each $n$ define
\[
 f_n=\frac{u_n+Ju_n}{\sqrt2},\qquad
 g_n=\frac{i(u_n-Ju_n)}{\sqrt2}.                             \tag{5.3}
\]
Then $f_n,g_n$ are orthonormal and fixed by $J$.  On their span, in the
ordered basis $(f_n,g_n)$, the matrix of $T$ is $i\lambda_nL_{12}$.

Suppose first that there are infinitely many positive eigenvalue pairs.
Group the pairs with indices $2k-1,2k$ into mutually orthogonal
four-dimensional $J$-invariant spaces $H_k$.  Relabeling within a block if
necessary, Lemma~\ref{lem:blocks} gives a skew-symmetric block $Y_k$ with
\[
 [Y_k^*,Y_k]=T|_{H_k},\qquad
 \norm{Y_k}\leq\sqrt2\max(\lambda_{2k-1},\lambda_{2k})^{1/2}.
                                                               \tag{5.4}
\]
The right-hand side tends to zero.

If there are finitely many positive eigenvalue pairs, group them in the same
way.  When their number is odd, one pair remains.  In that case $T$ has
finite rank, hence $\ker T$ is infinite-dimensional and $J$-invariant.
The restriction of $J$ to $\ker T$ has a fixed unit vector $h$: for a nonzero
$x\in\ker T$, either $x+Jx$ is nonzero and can be normalized, or $ix$ is
$J$-fixed after normalizing.  Adjoin $h$ to the remaining pair and apply the
three-dimensional part of Lemma~\ref{lem:blocks}.  This completes the finite
block decomposition in all cases.

Let $H_0$ be the orthogonal complement of the constructed blocks.  It is a
$J$-invariant subspace of $\ker T$.  Define
\[
 Y=0|_{H_0}\oplus\bigoplus_kY_k.                             \tag{5.5}
\]
Each block is skew-symmetric in a $J$-real basis, so (2.2) gives
$Y=-JY^*J$.  The block norms tend to zero in the only infinite case;
therefore the direct sum in (5.5) is compact.  Thus $Y\in\ooJ$.  Finally,
blockwise computation gives
\[
 [Y^*,Y]=0|_{H_0}\oplus\bigoplus_k[Y_k^*,Y_k]=T.
\]
Taking the supremum in the block estimate proves (5.1).
\end{proof}

\begin{corollary}[spectral characterization]\label{cor:spectrum}
A real null multiset is the eigenvalue multiset of a self-commutator
$[Y^*,Y]$ with $Y\in\ooJ$ if and only if every nonzero $\lambda$ and
$-\lambda$ occur with equal multiplicity.  Equivalently, after relabeling,
the nonzero terms have the form
\[
 \lambda_1,-\lambda_1,\lambda_2,-\lambda_2,\ldots,
 \qquad \lambda_n>0,quad \lambda_n\longrightarrow0,        \tag{5.6}
\]
with any allowed zero multiplicity.
\end{corollary}

\begin{proof}
Necessity follows from (2.1), because every $[Y^*,Y]$ is self-adjoint and
belongs to the Lie algebra $\ooJ$.  Conversely, realize (5.6) by the compact
block-diagonal operator whose two-dimensional $J$-real blocks are
$i\lambda_nL_{12}$, and apply Theorem~\ref{thm:main}.
\end{proof}

\section{Verification and novelty limits}

The identities in Lemma~\ref{lem:blocks} were checked both symbolically and
in 1,000 randomized orthogonally conjugated instances; see
\texttt{code/verify\_blocks.py} and \texttt{verification.md}.  These checks
are sanity tests, not a substitute for the displayed algebraic proof.

A bounded novelty check was performed on 19 July 2026.  The run's registry,
solution, attempt, and proof-gap indexes were searched for arXiv:1303.4844,
the exact title, ``type B'', ``orthogonal Lie algebra'', and
``self-commutator''.  Primary-source arXiv/web searches used the exact source
question, the phrase ``type (B) self-commutator'', the equation $[X^*,X]=A$
in the compact orthogonal algebra, the source authors, and citing/structural
work on the orthogonal operator Lie algebra.  No later paper explicitly
answering this type-(B) question was located.  This is a bounded search, not a
claim of exhaustive historical priority.

The theorem addresses only the type-(B) part of Question 4.10.  It does not
address the source paper's separate questions about positive single
commutators of compact operators, distinct eigenvalues, or block-size control.
