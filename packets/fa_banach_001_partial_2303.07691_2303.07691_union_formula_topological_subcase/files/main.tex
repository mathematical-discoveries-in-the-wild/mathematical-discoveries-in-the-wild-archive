\title{The Union Formula for \texorpdfstring{$(I)$}{(I)}-Envelopes:\\
Verification, Literature Status, and an Affine-Union Theorem}
\author{Research note prepared from Question~5.1 of Kalenda--Raja}
\date{June 22, 2026}

\begin{document}
\maketitle

\begin{abstract}
Kalenda and Raja asked whether
\[
 \Ienv(A\cup B)=\Ienv(A)\cup\Ienv(B)
\]
whenever $A,B\subset X^*$ are convex and $A\cup B$ is convex.  We first
check the proposed partial argument.  It is correct, but its topological
part is already exactly the implication $(1)\Rightarrow(2)$ in their
Theorem~4.3; combined with their Theorem~3.3 it gives the formula when $X$
contains no isomorphic copy of $\ell^1$.  A literature search through June
22, 2026 did not locate a full proof or a counterexample, and the current
arXiv abstract still describes the underlying non-locally-convex topological
problem as open.

We then record further positive results.  The formula is automatic when the
union is already $(I)$-closed (in particular, when it is
$\sigma(X^*,B_1(X))$-closed), and it holds whenever $A\cup B$ is norm
separable.  More substantially, it holds in every Banach space whenever
$A\cup B$ is an affine subspace of $X^*$ (not necessarily closed).  The latter is proved from the sequential separation criterion for
$(I)$-envelopes and a self-contained geometric lemma: two proper closed
convex sets covering a real Banach space must be two oppositely oriented
parallel half-spaces, possibly with an overlapping slab.  Consequently, any
counterexample to Question~5.1 must occur in a space containing $\ell^1$,
must have a non-norm-separable union, and cannot have affine union.
\end{abstract}

\section{The question and the outcome of the check}

Let $X$ be a real or complex Banach space.  For $S\subset X^*$, its
$(I)$-envelope is
\[
\Ienv(S)=
\bigcap\left\{
 \overline{\co\!\left(\bigcup_{n=1}^{\infty}
       \overline{\co S_n}^{\,w^*}\right)}^{\|\cdot\|}:
 S=\bigcup_{n=1}^{\infty}S_n
\right\}.
\]
The open problem is the following \cite[Question~5.1]{KR}.

\medskip
\noindent\textbf{Union question.}
If $A,B\subset X^*$ are convex and $A\cup B$ is convex, must
\[
\tag{UF}\label{UF}
 \Ienv(A\cup B)=\Ienv(A)\cup\Ienv(B)
\]
hold?

The inclusion from right to left always holds, by monotonicity of the
$(I)$-envelope.  The issue is the reverse inclusion.

\begin{proposition}[The proposed topological argument is correct]
\label{prop:topological}
Suppose there is a topology $\tau$ on $X^*$ such that
\[
 \Ienv(S)=\overline{\co S}^{\,\tau}
 \qquad(S\subset X^*).
\]
Then \eqref{UF} holds for every admissible pair $A,B$.  In particular,
\eqref{UF} holds if $X$ contains no isomorphic copy of $\ell^1$.
\end{proposition}

\begin{proof}
Put $C=A\cup B$.  Since $A,B,C$ are convex,
\[
\Ienv(C)=\overline C^{\,\tau}
=\overline{A\cup B}^{\,\tau}
=\overline A^{\,\tau}\cup\overline B^{\,\tau}
=\Ienv(A)\cup\Ienv(B).
\]
If $X$ contains no copy of $\ell^1$, Kalenda--Raja's Theorem~3.3 gives,
for every $S\subset X^*$,
\[
 \Ienv(S)=\overline{\co S}^{\,\sigma(X^*,B_1(X))},
\]
so the first part applies.
\end{proof}

\begin{remark}[Novelty of the submitted partial result]
The first assertion of Proposition~\ref{prop:topological} is not a new
lemma: it is precisely the proof of $(1)\Rightarrow(2)$ in
\cite[Theorem~4.3]{KR}.  The ``no $\ell^1$'' conclusion is the immediate
combination of that implication with \cite[Theorem~3.3]{KR}.  The submitted
partial result is therefore mathematically correct, including its use of
Theorem~3.3 for arbitrary subsets, but it should be presented as a useful
corollary already contained in the source paper rather than as a novel
result.
\end{remark}

\section{Sequential separation and elementary invariances}

All convex and affine notions below are over $\R$.  In the complex case we
use real parts, exactly as in \cite[Lemma~2.1]{KR}.

For a bounded sequence $\mathbf{x}=(x_n)\subset B_X$, set
\[
 p_{\mathbf{x}}(x^*)=\limsup_{n\to\infty}\Ree x^*(x_n),
 \qquad x^*\in X^*.
\]
This is a finite, positively homogeneous, subadditive, norm-Lipschitz
function.

\begin{lemma}[Sequential separation criterion \cite{Kalenda2007,KR}]
\label{lem:separation}
Let $S\subset X^*$ and $\eta\in X^*$.  Then $\eta\notin\Ienv(S)$ if and
only if there is a sequence $(x_n)\subset B_X$ such that
\[
\tag{3.1}\label{strongsep}
 \sup_{s\in S}\limsup_{n\to\infty}\Ree s(x_n)
 <\inf_{n\in\N}\Ree\eta(x_n).
\]
Equivalently, one may replace the right-hand side by
$\liminf_n\Ree\eta(x_n)$ or by $\limsup_n\Ree\eta(x_n)$.
\end{lemma}

We will repeatedly use two simple consequences.

\begin{lemma}[Norm closure and translation]
\label{lem:invariance}
For every $S\subset X^*$ and $u\in X^*$,
\[
 \Ienv(S)=\Ienv(\overline S^{\|\cdot\|}),
 \qquad
 \Ienv(S+u)=\Ienv(S)+u.
\]
\end{lemma}

\begin{proof}
For the first assertion, $p_{\mathbf{x}}$ is norm continuous and in fact
\[
 |p_{\mathbf{x}}(x^*)-p_{\mathbf{x}}(y^*)|
 \leq \|x^*-y^*\|.
\]
Thus its supremum on $S$ equals its supremum on the norm closure of $S$,
and Lemma~\ref{lem:separation} gives the equality of the envelopes.
Translation invariance follows directly from the defining countable-cover
formula: translation commutes with convex hulls, weak-star closure, norm
closure, and intersections.
\end{proof}

\section{Further immediate positive cases}

\begin{proposition}[$(I)$-closed union]
\label{prop:Iclosed}
Let $A,B\subset X^*$ be convex and let $C=A\cup B$ be convex.  If
$\Ienv(C)=C$, then \eqref{UF} holds.  In particular, this is true whenever
$C$ is $\sigma(X^*,B_1(X))$-closed, and hence whenever $C$ is weak-star
closed.
\end{proposition}

\begin{proof}
Every point of $C=A\cup B$ belongs to $\Ienv(A)\cup\Ienv(B)$.  Conversely,
monotonicity gives
\[
 \Ienv(A)\cup\Ienv(B)\subset\Ienv(C)=C.
\]
For the sufficient condition, Kalenda--Raja's Proposition~2.2 gives
\[
 \Ienv(S)\subset
 \overline{\co S}^{\,\sigma(X^*,B_1(X))}
 \qquad(S\subset X^*).
\]
Thus a convex $\sigma(X^*,B_1(X))$-closed set $C$ satisfies
$\Ienv(C)=C$.  Since $X\subset B_1(X)$, weak-star closedness is an even
stronger sufficient condition for a convex set.
\end{proof}

\begin{proposition}[Norm-separable union]
\label{prop:separable}
If $A,B\subset X^*$ are convex, $A\cup B$ is convex, and $A\cup B$ is norm
separable, then \eqref{UF} holds.
\end{proposition}

\begin{proof}
For every norm-separable set $S\subset X^*$ one has
\[
 \Ienv(S)=\overline{\co S}^{\|\cdot\|};
\]
see \cite[Remark~1.1(ii)]{Kalenda2007} or
\cite[Proposition~4.4(b)]{KR}.  Since $A$, $B$, and $C=A\cup B$ are convex
and norm separable,
\[
 \Ienv(C)=\overline C^{\|\cdot\|}
 =\overline{A\cup B}^{\|\cdot\|}
 =\overline A^{\|\cdot\|}\cup\overline B^{\|\cdot\|}
 =\Ienv(A)\cup\Ienv(B).
\]
\end{proof}

\section{Geometry of a two-set convex cover}

The key geometric input for the affine-union theorem is elementary but
useful in its own right.

For a nonempty closed convex set $P$ in a real Banach space $V$, write
\[
 \rec(P)=\{v\in V: P+[0,\infty)v\subset P\}
\]
for its recession cone.

\begin{lemma}[One ray determines a recession direction]
\label{lem:ray}
Let $P\subset V$ be nonempty, closed, and convex.  If $p_0+tv\in P$ for all
$t\geq0$ for one $p_0\in P$, then $v\in\rec(P)$.
\end{lemma}

\begin{proof}
Fix $p\in P$ and $t\geq0$.  For $s>t$, convexity gives
\[
 \left(1-\frac ts\right)p+\frac ts(p_0+sv)
 =p+tv+\frac ts(p_0-p)\in P.
\]
Letting $s\to\infty$ and using closedness yields $p+tv\in P$.
\end{proof}

\begin{lemma}[Two closed convex cones covering a Banach space]
\label{lem:cones}
Let $C,D$ be proper closed convex cones in a real Banach space $V$, and
suppose $C\cup D=V$.  Then there is a nonzero $g\in V^*$ such that, after
interchanging $C$ and $D$ if necessary,
\[
 C=\{v:g(v)\leq0\},\qquad D=\{v:g(v)\geq0\}.
\]
\end{lemma}

\begin{proof}
We first claim that
\[
\tag{5.1}\label{opposite}
 x\in C\setminus D\quad\Longrightarrow\quad -x\in D\setminus C.
\]
Indeed, if also $-x\in C$, then the whole line $\R x$ lies in $C$.  Choose
$y\in D\setminus C$, which is possible because $C$ is proper and
$C\cup D=V$.  For every $t\in\R$, the point $y+tx$ cannot lie in $C$;
otherwise $y=(y+tx)-tx\in C$.  Hence $y+tx\in D$ for every $t$.  For
$t>0$, $(y+tx)/t\in D$, and closedness gives $x\in D$ as $t\to\infty$, a
contradiction.  This proves \eqref{opposite}; the symmetric statement holds
with $C,D$ interchanged.

Set $H=C\cap D$.  It is a closed linear subspace.  It is already a convex
cone; if $h\in H$ and $-h$ belonged to exactly one of $C,D$, then
\eqref{opposite} would force $h$ to belong to exactly the other one, a
contradiction.  Thus $-h\in H$.

Let $q:V\to V/H$ be the quotient map.  Since $C+H=C$ and $D+H=D$, the sets
$q(C)$ and $q(D)$ are closed convex cones in $V/H$.  They cover $V/H$ and
intersect only at $0$.  Moreover, by \eqref{opposite},
$q(D)=-q(C)$.  Consequently
\[
 (V/H)\setminus\{0\}
 =\bigl(q(C)\setminus\{0\}\bigr)
  \mathbin{\dot\cup}
  \bigl(q(D)\setminus\{0\}\bigr)
\]
is a separation into two nonempty relatively closed sets.  A real normed
space of dimension at least two remains path connected after deleting the
origin: join two nonzero points by a segment unless they are opposite, and
in the opposite case use an intermediate vector outside their line.  Hence
$\dim(V/H)=1$.

Thus $H$ is a closed hyperplane.  Choose a nonzero $g\in V^*$ with
$H=\ker g$.  The two quotient cones are the two half-lines in $\R$, so after
reversing the sign of $g$ if needed, the asserted formulas follow.
\end{proof}

\begin{proposition}[Slab structure]
\label{prop:slab}
Let $P,Q$ be proper nonempty closed convex subsets of a real Banach space
$V$, and suppose $P\cup Q=V$.  Then there are a nonzero $g\in V^*$ and
numbers $\beta\leq\alpha$ such that, after interchanging $P,Q$ if necessary,
\[
\tag{5.2}\label{slabs}
 P=\{v\in V:g(v)\leq\alpha\},
 \qquad
 Q=\{v\in V:g(v)\geq\beta\}.
\]
\end{proposition}

\begin{proof}
Because $V$ is connected and $P,Q$ are closed proper sets covering $V$, one
has $P\cap Q\neq\varnothing$.  Translate by a point of the intersection, so
that $0\in P\cap Q$.

Fix $v\in V$.  The sets
\[
 I_P=\{t\geq0:tv\in P\},\qquad
 I_Q=\{t\geq0:tv\in Q\}
\]
are closed intervals containing $0$, and $I_P\cup I_Q=[0,\infty)$.  At
least one is unbounded, hence equals $[0,\infty)$.  By
Lemma~\ref{lem:ray}, $v\in\rec(P)\cup\rec(Q)$.  Therefore
\[
 \rec(P)\cup\rec(Q)=V.
\]
Neither recession cone is all of $V$, since $0\in P\cap Q$ and both $P,Q$
are proper.  Lemma~\ref{lem:cones} supplies a nonzero $g\in V^*$ such that
\[
 \rec(P)=\{g\leq0\},\qquad \rec(Q)=\{g\geq0\}.
\]
In particular, $H=\ker g$ is contained in the lineality space of each set,
so $P+H=P$ and $Q+H=Q$.

Hence membership in $P$ depends only on the scalar $g(v)$.  The set $g(P)$
is a nonempty closed convex subset of $\R$ (closedness follows because $P$
is saturated with respect to the quotient map $V\to V/H$).  It is downward
closed, because every direction with negative $g$-value belongs to
$\rec(P)$.  Since $P$ is proper, $g(P)=(-\infty,\alpha]$ for some finite
$\alpha$.  Similarly, $g(Q)=[\beta,\infty)$ for some finite $\beta$.
The covering property gives $\beta\leq\alpha$, proving \eqref{slabs}.
\end{proof}

\section{The affine-union theorem}

\begin{theorem}[Affine-union formula]
\label{thm:affine}
Let $X$ be any real or complex Banach space.  Let $A,B\subset X^*$ be
convex, and suppose that $A\cup B$ is an affine subspace of the underlying
real vector space of $X^*$.  Then
\[
 \Ienv(A\cup B)=\Ienv(A)\cup\Ienv(B).
\]
The affine subspace need not be norm closed.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:invariance}, replacing $A,B$ and $A\cup B$ by their norm
closures does not change any of the three $(I)$-envelopes.  Since norm
closure commutes with finite unions,
\[
 \overline{A\cup B}^{\|\cdot\|}
 =\overline A^{\|\cdot\|}\cup\overline B^{\|\cdot\|},
\]
and the closure of an affine subspace is affine.  Translation invariance
therefore reduces the problem to the following situation: $V\subset X^*$
is a norm-closed real linear subspace, and $P,Q\subset V$ are closed convex
sets with $P\cup Q=V$.

If $P=V$ or $Q=V$, the result is immediate.  Otherwise
Proposition~\ref{prop:slab} gives $g\in V^*\setminus\{0\}$ and
$\beta\leq\alpha$ such that
\[
 P=\{v\in V:g(v)\leq\alpha\},
 \qquad
 Q=\{v\in V:g(v)\geq\beta\}.
\]
Choose $u\in V$ with $g(u)=1$, and put $H=\ker g$.

We prove the nontrivial inclusion by contraposition.  Suppose
\[
 \eta\notin\Ienv(P)\cup\Ienv(Q).
\]
Apply Lemma~\ref{lem:separation} to $P$.  There is $(x_n)\subset B_X$ with
\[
 M_P:=\sup_{p\in P}\limsup_n\Ree p(x_n)
 <\inf_n\Ree\eta(x_n).
\]
Write $p_{\mathbf{x}}(v)=\limsup_n\Ree v(x_n)$.  Fix $p_0\in P$.
For every $h\in H$ and $t>0$, $p_0\pm th\in P$, hence subadditivity gives
\[
 t p_{\mathbf{x}}(\pm h)
 =p_{\mathbf{x}}(\pm th)
 \leq p_{\mathbf{x}}(p_0\pm th)+p_{\mathbf{x}}(-p_0)
 \leq M_P+p_{\mathbf{x}}(-p_0).
\]
Letting $t\to\infty$ shows
$p_{\mathbf{x}}(h)\leq0$ and $p_{\mathbf{x}}(-h)\leq0$; since
$0\leq p_{\mathbf{x}}(h)+p_{\mathbf{x}}(-h)$, both are zero.  Thus
\[
\tag{6.1}\label{Hvanishx}
 \Ree h(x_n)\longrightarrow0\qquad(h\in H).
\]
Also $p_0-tu\in P$ for $t\geq0$, so the same argument gives
$p_{\mathbf{x}}(-u)\leq0$, or
$\liminf_n\Ree u(x_n)\geq0$.

Pass to a subsequence on which
\[
 \Ree u(x_n)\to r\geq0,
 \qquad
 \Ree\eta(x_n)\to s.
\]
For $p\in P$, the vector $p-g(p)u$ lies in $H$, so by
\eqref{Hvanishx}
\[
 \Ree p(x_n)\longrightarrow r g(p).
\]
Consequently the supremum of the new limsups over $P$ is $r\alpha$.  Passing
to a subsequence cannot increase any individual limsup, and its
$\eta$-limit is at least the old infimum; hence
\[
\tag{6.2}\label{ineqP}
 s>r\alpha.
\]

Apply the same argument to a separating sequence $(y_n)\subset B_X$ for
$Q$.  Since $H$ is again the lineality space while $u\in\rec(Q)$, after
passing to a subsequence we obtain
\[
\tag{6.3}\label{limitsQ}
 \Ree h(y_n)\to0\ (h\in H),\qquad
 \Ree u(y_n)\to t\leq0,\qquad
 \Ree\eta(y_n)\to v>t\beta.
\]

If $r=0$, then \eqref{Hvanishx} and the decomposition
$c=g(c)u+(c-g(c)u)$ show that $\Ree c(x_n)\to0$ for every $c\in V$, whereas
$s>0$ by \eqref{ineqP}.  Lemma~\ref{lem:separation} gives
$\eta\notin\Ienv(V)$.  The case $t=0$ follows similarly from
\eqref{limitsQ}.

It remains to consider $r>0$ and $t<0$.  Define
\[
 z_n=(-t)x_n+r y_n,
 \qquad
 w_n=\frac{z_n}{-t+r}\in B_X.
\]
For each $c\in V$, using $c=g(c)u+(c-g(c)u)$ and
\eqref{Hvanishx}--\eqref{limitsQ},
\[
 \Ree c(z_n)\longrightarrow (-t)r g(c)+rt g(c)=0.
\]
On the other hand, by \eqref{ineqP} and \eqref{limitsQ},
\[
 \lim_n\Ree\eta(z_n)
 =(-t)s+rv
 >(-t)r\alpha+rt\beta
 =rt(\beta-\alpha)\geq0,
\]
because $r>0$, $t<0$, and $\beta\leq\alpha$.  After division by
$-t+r$, the $\eta$-limit remains strictly positive, while every element of
$V$ has real part tending to zero on $(w_n)$.  Lemma~\ref{lem:separation}
again yields $\eta\notin\Ienv(V)$.

We have proved
\[
 \eta\notin\Ienv(P)\cup\Ienv(Q)
 \quad\Longrightarrow\quad
 \eta\notin\Ienv(V),
\]
which is the desired reverse inclusion.  The forward inclusion follows from
monotonicity.
\end{proof}

\begin{corollary}
No counterexample to Question~5.1 can have $A\cup B$ equal to a linear
subspace, a hyperplane, or more generally any affine subspace of $X^*$.
In particular, the tempting strategy of splitting a hyperplane
$\ker x^{**}$ into two convex pieces cannot produce a counterexample, even
when $x^{**}\notin B_1(X)$ and $\Ienv(\ker x^{**})=X^*$.
\end{corollary}

\section{What a counterexample would now have to look like}

Combining the source theorem with Propositions~\ref{prop:Iclosed} and
\ref{prop:separable} and Theorem~\ref{thm:affine} gives four necessary
conditions for a counterexample:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.4em]
\item $X$ must contain an isomorphic copy of $\ell^1$;
\item $A\cup B$ must be non-norm-separable;
\item $A\cup B$ must be genuinely non-affine;
\item $A\cup B$ must fail to be $(I)$-closed (in particular, it cannot be
$\sigma(X^*,B_1(X))$-closed or weak-star closed).
\end{enumerate}
These restrictions eliminate several of the most natural constructions
based on separable subspaces, hyperplanes, or opposing half-spaces.

It is useful to spell out the remaining technical obstruction.  Given a
bounded sequence $(x_n)\subset X$, let $K$ be its weak-star cluster set in
$B_{X^{**}}$.  Then
\[
 p_{\mathbf{x}}(x^*)
 =\max_{x^{**}\in K}\Ree x^{**}(x^*).
\]
A strong separating sequence for $A$ therefore produces a compact cluster
set all of whose points separate $\eta$ from $A$ with a uniform gap; a
second sequence does the analogous job for $B$.  To separate from
$A\cup B$, one would like a sequence whose cluster set lies in the
intersection of the two separator regions.  In a nonmetrizable weak-star
compact space, however, cluster sets generated by sequences are not stable
under arbitrary intersections.  This is the point at which the obvious
``combine the two witnesses'' argument breaks down.  The affine proof above
works because the slab geometry forces the restrictions of the two
sequences to be one-dimensional, allowing an explicit cancellation.

The general convex-union problem has no comparable one-dimensional
reduction in the argument currently known here.  Thus the results above are
strictly partial and do not settle Question~5.1.

\section{Conclusion}

The submitted proof is valid but already contained in the source paper's
Theorem~4.3, with its no-$\ell^1$ consequence supplied by Theorem~3.3.  No
full solution or counterexample was located in the literature search.  The
newest rigorous progress obtained here is the affine-union theorem,
together with the norm-separable and $(I)$-closed union observations.  These results narrow any
future counterexample search to a nonseparable, non-affine convex union in
the dual of a Banach space containing $\ell^1$.
