\section*{Source Problem}

The source paper is Jinghao Huang, Karimbergen Kudaybergenov, and Rui Liu,
\emph{The ball-covering property of von Neumann algebras and noncommutative
symmetric spaces}, arXiv:2311.14257.  After proving that a commutative unital
\(C^*\)-algebra \(C(K)\) has BCP/UBCP exactly when \(K\) has a countable
\(\pi\)-basis, the authors ask for a noncommutative version:

\section*{Definitions}

A family \(\mathcal U\) of nonempty open subsets of a topological space \(X\)
is \(\pi\)-dense if every nonempty open subset of \(X\) contains a member of
\(\mathcal U\).  Equivalently, \(\mathcal U\) is a \(\pi\)-basis.

For an open set \(O\subset\Prim(A)\), let \(A_O\) denote the corresponding
closed two-sided ideal of \(A\).

\section*{Statement}

\begin{theorem}[\(\pi\)-dense atomic chart criterion]
\label{thm:pidense}
Let \(A\) be a \(C^*\)-algebra.  Suppose there is a countable \(\pi\)-dense
family \((O_n)\) of open subsets of \(\Prim(A)\) such that
\[
        A_{O_n}\cong C_0(O_n,\Kcal(H_n))
\]
for separable nonzero Hilbert spaces \(H_n\).  Then \(A\) has UBCP.
\end{theorem}

\begin{corollary}[essential Fell ideal criterion]
\label{cor:essential-fell}
Let \(A\) be a \(C^*\)-algebra.  Suppose \(A\) has an essential ideal \(J\)
which is Fell, and suppose every irreducible representation of \(J\) acts on
a separable Hilbert space.  Then
\[
        A\text{ has BCP}
        \quad\Longleftrightarrow\quad
        A\text{ has UBCP}
        \quad\Longleftrightarrow\quad
        \Prim(A)\text{ has a countable }\pi\text{-basis}.
\]
\end{corollary}

\section*{Idea of the Proof}

The atomic multiplier patch proof does not need good charts everywhere.  For
each \(a\in S_A\), the set
\[
        W_a=\{P\in\Prim(A):\|a+P\|>3/4\}
\]
is nonempty and open.  A countable \(\pi\)-dense family of atomic charts gives
one chart \(O_n\subset W_a\), and the rank-one coefficient construction from
the Fell packet covers \(a\) using a centre supported in that chart.

If \(J\) is essential, then \(\Prim(J)\) is a dense open subset of
\(\Prim(A)\).  Therefore any countable \(\pi\)-basis of \(\Prim(A)\) can be
refined inside \(\Prim(J)\).  Since \(J\) is Fell with separable irreducible
representations, its spectrum is locally Hausdorff and its Hausdorff patches
are continuous trace with separable fibres.  Local triviality then gives the
required \(\pi\)-dense atomic charts.

\section*{Proof}

\begin{lemma}[rank-one coefficient estimate]
\label{lem:rankone}
Fix \(0<\theta<1\).  There are constants \(M>1\) and \(\rho<M\) such that,
whenever \(T\in B(H)\) is a contraction, \(\xi,\eta\in H\) are unit vectors,
and
\[
        \operatorname{Re}\langle T\xi,\eta\rangle\ge\theta,
\]
then, for \(R u=\langle u,\xi\rangle\eta\),
\[
        \sup_{0\le t\le M}\|T-tR\|\le\rho .
\]
\end{lemma}

\begin{proof}
This is Lemma 1 of the atomic multiplier chart packet
\cite{lane19atomic}.  The proof is a two-region estimate on a unit vector
\(x=\lambda\xi+v\), \(v\perp\xi\).  If \(|\lambda|\) is bounded away from
one, the large rank-one term is already smaller than \(t\) by a fixed factor;
if \(|\lambda|\) is close to one, the positive coefficient
\(\operatorname{Re}\langle T\xi,\eta\rangle\) subtracts a fixed linear amount
from the square norm.  For large \(M\) this gives a uniform
\(\rho<M\).
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:pidense}]
Fix \(\theta=1/2\), choose \(M,\rho\) from Lemma~\ref{lem:rankone}, and put
\[
        r=\frac{M+\rho}{2}.
\]
Thus \(\rho<r<M\).

For each \(n\), fix an isomorphism
\[
        \Phi_n:A_{O_n}\to C_0(O_n,\Kcal(H_n)),
\]
a countable dense set \(D_n\) in the unit sphere of \(H_n\), and a countable
family \(\mathcal B_n\subset C_c(O_n)\), \(0\le h\le1\), \(\|h\|=1\), whose
supports are cofinal in nonempty open subsets of \(O_n\).  These bump
families exist because the \(O_n\)'s are locally compact Hausdorff and the
countable \(\pi\)-dense family restricts to a countable \(\pi\)-basis on each
\(O_n\).

For each \(n\), \(h\in\mathcal B_n\), and \(\xi,\eta\in D_n\), define
\[
        c(n,h,\xi,\eta)
        =
        \Phi_n^{-1}\bigl(Mh(\eta\otimes\xi^*)\bigr)
        \in A_{O_n}\subset A.
\]
These centres are countable and have norm \(M\).

Let \(a\in S_A\), and set
\[
        W_a=\{P\in\Prim(A):\|a+P\|>3/4\}.
\]
This is nonempty open, so choose \(O_n\subset W_a\).  Pick \(P_0\in O_n\).
In the multiplier representation of \(a\) over
\[
        A_{O_n}\cong C_0(O_n,\Kcal(H_n)),
\]
let \(T_P\in B(H_n)\) denote the fibre multiplier.  Then
\(\|T_{P_0}\|>3/4\), so choose unit vectors \(\xi_0,\eta_0\) with
\[
        \operatorname{Re}\langle T_{P_0}\xi_0,\eta_0\rangle>2/3.
\]
Approximate them by \(\xi,\eta\in D_n\) so that the coefficient remains
larger than \(1/2\).

Multiplier fields are strict-continuous, hence fixed scalar matrix
coefficients are continuous.  Therefore
\[
        \operatorname{Re}\langle T_P\xi,\eta\rangle\ge1/2
\]
on some nonempty open \(N\subset O_n\).  Choose \(h\in\mathcal B_n\) with
\(\supp h\subset N\), and put \(c=c(n,h,\xi,\eta)\).

For \(P\in\supp h\), Lemma~\ref{lem:rankone}, applied with
\(t=Mh(P)\), gives
\[
        \|a+P-(c+P)\|\le\rho<r.
\]
For \(P\notin\supp h\), the centre vanishes in the primitive quotient at
\(P\), so
\[
        \|a+P-(c+P)\|=\|a+P\|\le1<r,
\]
after increasing \(M\), if necessary.  Taking the supremum over primitive
quotients gives \(\|a-c\|<r\).  The countably many balls
\[
        B(c(n,h,\xi,\eta),r)
\]
therefore cover \(S_A\), and since every centre has norm \(M>r\), \(A\) has
UBCP.
\end{proof}

\begin{lemma}[essential ideals give dense primitive support]
\label{lem:essential}
For an ideal \(J\triangleleft A\), \(J\) is essential if and only if
\(\Prim(J)\), identified with the open subset
\[
        \{P\in\Prim(A):P\not\supset J\},
\]
is dense in \(\Prim(A)\).
\end{lemma}

\begin{proof}
The complement is the hull of \(J\).  If this hull had nonempty interior, the
corresponding nonzero ideal would be disjoint from \(J\), so \(J\) would not
be essential.  Conversely, if a nonzero ideal is disjoint from \(J\), its
primitive support gives a nonempty open subset of the hull of \(J\).  Thus
essentiality is equivalent to the hull having empty interior, i.e. to
\(\Prim(J)\) being dense.
\end{proof}

\begin{proof}[Proof of Corollary~\ref{cor:essential-fell}]
If \(A\) has BCP, the primitive \(\pi\)-base obstruction
\cite{lane19primitive} gives a countable \(\pi\)-basis for \(\Prim(A)\).
UBCP implies BCP, so it remains to prove UBCP from the countable
\(\pi\)-basis assumption.

By Lemma~\ref{lem:essential}, \(\Prim(J)\) is dense open in \(\Prim(A)\).
Since \(J\) is Fell, \(\Prim(J)\) is locally Hausdorff; on Hausdorff open
patches, Fell algebras have continuous trace \cite{CHR}.  Local triviality of
continuous-trace algebras and separability of the irreducible representations
give, inside every nonempty open subset of \(\Prim(J)\), a nonempty open
patch \(O\) such that
\[
        J_O=A_O\cong C_0(O,\Kcal(H))
\]
for separable nonzero \(H\).

Now refine a countable \(\pi\)-basis of \(\Prim(A)\) by these patches inside
the dense open set \(\Prim(J)\).  The resulting countable family is
\(\pi\)-dense in \(\Prim(A)\) and satisfies the hypotheses of
Theorem~\ref{thm:pidense}.  Hence \(A\) has UBCP.
\end{proof}

\section*{Scope and Limitations}

This theorem is stronger than the Fell-algebra packet because \(A\) itself
need not be Fell, type I, liminal, or separable.  The quotient by the
essential Fell ideal can be arbitrary.  The theorem does not handle the case
where the largest separable-Fell ideal is not essential; then the complement
has nonempty primitive interior and must be analyzed as a genuine remaining
piece.
