\section{Source question}

Garcia-Lirola, Grelier, Martinez-Cervantes, and Rueda Zoca ask the following
on page 13 of \cite{GGMR}.

\begin{quote}
\textbf{Question 3.9.} Is $x\otimes y$ a preserved extreme point of
$B_{X\pt Y}$ whenever $x$ and $y$ are preserved extreme points of $B_X$ and
$B_Y$?
\end{quote}

\begin{figure}[h]
  \centering
  \includegraphics[width=\textwidth]{figures/open_problem_crop.png}
  \caption{Questions 3.9 and 3.10 in arXiv:2211.13559, PDF page 13.}
\end{figure}

All spaces below are real. Recall that $x\in B_X$ is \emph{preserved
extreme} if its canonical image is an extreme point of $B_{X^{**}}$.

\section{Idea of the counterexample}

Weak exposure and norm exposure have very different tensor behavior.  We
renorm $\ell_2$ so that the points
\[
 (1-\varepsilon_n)e_0+e_n
 \quad\hbox{and}\quad
 (1-\varepsilon_n)e_0-e_n
\]
belong to the unit ball, where $\varepsilon_n\downarrow0$.  They both converge
weakly, but not in norm, to $e_0$.  The $e_0$-coordinate nevertheless weakly
strongly exposes $e_0$: near-maximizers must place asymptotically all their
mass on high-index generators, and those generators converge weakly to
$e_0$.  This already implies that $e_0$ is preserved extreme.

Tensoring preserves the hidden norm-sized perturbations.  Correlate the
positive perturbation in the second factor once with the positive perturbation
and once with the negative perturbation in the first factor.  The two tensor
sequences have distinct weak-star cluster points, detected by the Hilbert
inner product, while their midpoint loses the perturbation and converges to
$e_0\otimes e_0$.  Hence $e_0\otimes e_0$ is not extreme in the tensor
bidual.

\section{The renorming}

Let $H=\ell_2(\{0,1,2,\ldots\})$ over $\R$, with orthonormal basis
$(e_n)_{n\geq0}$.  For $n\geq1$, put
\[
 \varepsilon_n=\frac1{n+1},
 \qquad
 p_n^{\sigma}=(1-\varepsilon_n)e_0+\sigma e_n
 \quad(\sigma\in\{-1,1\}).
\]
Define
\[
 G=\frac12 B_H\ \cup\
   \{\tau p_n^{\sigma}:n\geq1,\ \sigma,\tau\in\{-1,1\}\},
 \qquad
 B=\overline{\operatorname{co}}(G),
 \tag{1}
\]
where closure is in the Hilbert norm.  Let $\lVert\cdot\rVert_X$ be the
Minkowski functional of $B$ and write $X=(H,\lVert\cdot\rVert_X)$.

\begin{lemma}\label{lem:norm}
The set $B$ is the closed unit ball of an equivalent norm on $H$, and
\[
 \frac1{\sqrt2}\lVert z\rVert_H
 \leq \lVert z\rVert_X
 \leq 2\lVert z\rVert_H
 \qquad(z\in H).
 \tag{2}
\]
Moreover, $e_0\in B$.
\end{lemma}

\begin{proof}
The set $G$ is symmetric, and every $p_n^\sigma$ has Hilbert norm at most
$\sqrt2$.  Hence
\[
 \frac12B_H\subseteq B\subseteq\sqrt2B_H.
\]
Thus $B$ is the closed unit ball of an equivalent norm and (2) follows.
Finally,
\[
 \frac{p_n^1+p_n^{-1}}2=(1-\varepsilon_n)e_0
 \longrightarrow e_0
\]
in Hilbert norm, so closedness gives $e_0\in B$.
\end{proof}

Let $f\in H^*$ be the zeroth-coordinate functional,
$f(z)=\langle z,e_0\rangle_H$.

\begin{lemma}\label{lem:preserved}
The functional $f$ has $\lVert f\rVert_{X^*}=1$, $f(e_0)=1$, and every net
$(z_\alpha)\subset B$ satisfying $f(z_\alpha)\to1$ converges weakly to
$e_0$.  Consequently, $e_0$ is a preserved extreme point of $B$.
\end{lemma}

\begin{proof}
On $\frac12B_H$ we have $f\leq\frac12$, while
\[
 f(p_n^\sigma)=1-\varepsilon_n,
 \qquad f(-p_n^\sigma)=-1+\varepsilon_n.
\]
Together with Lemma~\ref{lem:norm}, this proves
$\sup_B f=f(e_0)=1$.

We prove the asserted weak convergence by a uniform slice estimate.  Fix
$h=(h_j)\in H$, viewed as the Hilbert functional $z\mapsto\langle z,h\rangle$,
and fix $\eta>0$.  Choose $N$ so large that
\[
 \varepsilon_n|h_0|+|h_n|<\eta
 \qquad(n>N).
 \tag{3}
\]
Call the generators $p_n^\sigma$ with $n>N$ good and all other elements of
$G$ bad.  Every good generator $g$ satisfies
\[
 |\langle g-e_0,h\rangle|<\eta.
 \tag{4}
\]
Every bad generator has a uniform deficit
\[
 1-f(g)\geq c_N:=\min\{1/2,\varepsilon_N\}>0.
 \tag{5}
\]
If $z=\sum_j\lambda_jg_j$ is a finite convex combination of generators and
$f(z)>1-\delta$, (5) shows that the total coefficient of its bad generators
is at most $\delta/c_N$.  Since
\[
 C_h:=\sup_{g\in G}|\langle g-e_0,h\rangle|<\infty,
\]
(4) gives
\[
 |\langle z-e_0,h\rangle|
 \leq \eta+\frac{\delta}{c_N}C_h.
 \tag{6}
\]
Approximation by finite convex combinations extends (6), with an arbitrarily
small error, to all $z\in B$.  Letting first $\delta\downarrow0$ and then
$\eta\downarrow0$ proves that $f(z_\alpha)\to1$ forces
$z_\alpha\to e_0$ weakly.

For completeness, suppose
\[
 e_0=\frac{x_1^{**}+x_2^{**}}2,
 \qquad x_1^{**},x_2^{**}\in B_{X^{**}}.
 \]
Evaluation by $f$ forces $f(x_i^{**})=1$.  By Goldstine's theorem, approximate
each $x_i^{**}$ weak-star by a net in $B$.  The preceding paragraph makes
that net converge weakly to $e_0$, and therefore its weak-star limit in
$X^{**}$ is $e_0$.  Thus $x_1^{**}=x_2^{**}=e_0$, proving preserved
extremality.
\end{proof}

\section{Failure in the projective tensor bidual}

\begin{theorem}\label{thm:counterexample}
For the equivalent renorming $X$ above, $e_0$ is a preserved extreme point of
$B_X$, but $e_0\otimes e_0$ is not a preserved extreme point of
$B_{X\pt X}$.  Therefore Question 3.9 has a negative answer.
\end{theorem}

\begin{proof}
Set
\[
 r_n=p_n^1=(1-\varepsilon_n)e_0+e_n,
 \qquad
 s_n=p_n^{-1}=(1-\varepsilon_n)e_0-e_n,
\]
and, in $Z=X\pt X$, define
\[
 U_n=r_n\otimes r_n,
 \qquad V_n=s_n\otimes r_n.
 \tag{7}
\]
Both $r_n$ and $s_n$ belong to $B_X$, so $U_n,V_n\in B_Z$.  By weak-star
compactness of $B_{Z^{**}}\times B_{Z^{**}}$, there is a common subnet along
which
\[
 U_n\overset{w^*}{\longrightarrow}U,
 \qquad V_n\overset{w^*}{\longrightarrow}V
 \qquad(U,V\in B_{Z^{**}}).
 \tag{8}
\]

The midpoint has the exact form
\[
 \frac{U_n+V_n}{2}
 =\frac{r_n+s_n}{2}\otimes r_n
 =(1-\varepsilon_n)e_0\otimes
   \bigl((1-\varepsilon_n)e_0+e_n\bigr).
 \tag{9}
\]
The dual $Z^*$ is the space of bounded bilinear forms on $X\times X$.  For
any such form $A$, evaluation of (9) is
\[
 (1-\varepsilon_n)^2A(e_0,e_0)
 +(1-\varepsilon_n)A(e_0,e_n)
 \longrightarrow A(e_0,e_0),
\]
because $(e_n)$ is weakly null in $X$ (the norm of $X$ is equivalent to the
Hilbert norm).  Hence (9) converges weak-star to $e_0\otimes e_0$.  Passing to
the subnet in (8) yields
\[
 e_0\otimes e_0=\frac{U+V}{2}.
 \tag{10}
\]

It remains to show that this midpoint is nontrivial.  The original Hilbert
inner product
\[
 Q(a,b)=\langle a,b\rangle_H
\]
is a bounded bilinear form on $X\times X$ by (2).  From (7),
\[
 Q(U_n)=(1-\varepsilon_n)^2+1\longrightarrow2,
 \qquad
 Q(V_n)=(1-\varepsilon_n)^2-1\longrightarrow0.
\]
Therefore $Q(U)=2$ and $Q(V)=0$, so $U\ne V$.  Equation (10) is a nontrivial
midpoint decomposition inside $B_{Z^{**}}$.  Thus $e_0\otimes e_0$ is not
preserved extreme.  Lemma~\ref{lem:preserved} verifies that both of its
factors satisfy the hypothesis of Question 3.9.
\end{proof}

\section{Limitations and novelty check}

The theorem completely answers Question 3.9 in the negative.  It does not
answer Question 3.10: the tensor in the counterexample is basic, and the
argument concerns failure of preserved extremality rather than existence of a
non-basic extreme tensor.

A bounded arXiv search on 22 July 2026 used the exact source-question phrase,
the source arXiv id, and combinations of ``preserved extreme'', ``projective
tensor'', ``counterexample'', and ``weak-strongly exposed''.  It located the
source and the later positive partial result \cite{APEP}, but no counterexample
or full resolution.  This supports, but does not certify, novelty.

\section{Verification and human-review focus}

The proof was checked at four sensitive points: equivalence of the new norm,
the uniform weak slice estimate (6), simultaneous weak-star compactness in
(8), and separation of the two limits by $Q$.  No numerical computation is
used.  Human review should focus first on Lemma~\ref{lem:preserved} and then on
the midpoint convergence in (9).
