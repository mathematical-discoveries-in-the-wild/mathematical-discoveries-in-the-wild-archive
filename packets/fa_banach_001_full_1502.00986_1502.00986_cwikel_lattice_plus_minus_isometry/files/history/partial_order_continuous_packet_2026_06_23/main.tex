\section*{Status}

\textbf{Candidate partial result, likely valid.}
This packet proves the isometry conjectured in Remark 4.1 of Cwikel's
paper \cite{Cwikel2015} for couples of complex Banach lattices of
measurable functions whose endpoint lattice norms are order continuous.
The unrestricted case of arbitrary complexified Banach lattices is not
claimed here.

\section*{Source Problem}

The source paper is Michael Cwikel, ``Some alternative definitions for
the `plus-minus' interpolation spaces
$\left\langle A_0,A_1\right\rangle_\theta$ of Jaak Peetre,''
arXiv:1502.00986v2 \cite{Cwikel2015}.  The relevant statement is Remark
4.1 on page 13.

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

The remark recalls that, for couples of complexified Banach lattices of
measurable functions on the same measure space, the plus-minus space
$\left\langle A_0,A_1\right\rangle_\theta$ is known to coincide with
Calderon's complex interpolation space $[A_0,A_1]_\theta$ up to
equivalent norms.  It then conjectures that the coincidence is isometric
when the plus-minus side is equipped with the continuous norm
$\left\langle A_0,A_1\right\rangle_{\theta,(0)}$.

\section*{Idea of the Proof}

Cwikel already proves the norm-one inclusion
$\PM\subset \Cplx$, so only the reverse norm inequality is needed.
For Banach lattices, the complex interpolation norm is given by the
Calderon product norm.  A Calderon factorization can be written in the
pointwise form
\[
 |f|\le b,\qquad b e^{-\theta s}\in X_0,\qquad
 b e^{(1-\theta)s}\in X_1 .
\]
The scalar parameter $s$ says where the mass of $f$ should be placed on
the interpolation line.  Put a narrow kernel in the $t$-variable centered
at $s(\omega)$:
\[
 u(t,\omega)=\alpha(\omega)b(\omega)k(t-s(\omega)),
 \qquad |\alpha|\le 1,\quad \alpha b=f.
\]
Then $\int u=f$, while testing against every $|\phi|\le 1$ gives the
pointwise endpoint estimates
\[
\left|\int e^{-\theta t}\phi(t)u(t)\,dt\right|
 \lesssim b e^{-\theta s},\qquad
\left|\int e^{(1-\theta)t}\phi(t)u(t)\,dt\right|
 \lesssim b e^{(1-\theta)s}.
\]
For finite-valued $s$ this is directly an admissible continuous
plus-minus representation.  For general $s$, truncate to bounded
$s$-bands and quantize.  Order continuity of the two endpoint lattice
norms makes the discarded tails small in both endpoint estimates, which
lets the finite-band argument pass to the limit.

\section*{Definitions}

Let $X_0$ and $X_1$ be complex Banach lattices of measurable functions
on a common measure space, and let $0<\theta<1$.  We write
$\PM$ for Cwikel's continuous plus-minus space and norm from
\cite[Section 3]{Cwikel2015}.  Thus an element $f$ is represented as
$f=\int_{-\infty}^{\infty}u(t)\,dt$, where $u$ is a locally piecewise
continuous $X_0\cap X_1$-valued function whose weighted improper
integrals are unconditionally convergent after multiplication by all
locally piecewise continuous scalar functions $\phi$ with
$\norm{\phi}_\infty\le 1$.  Its $J$-norm is
\[
\norm{u}_J
=\max_{j=0,1}\sup_{\phi}
\left\|\int_{-\infty}^{\infty} e^{(j-\theta)t}\phi(t)u(t)\,dt
\right\|_{X_j},
\]
and $\norm{f}_{\PM}$ is the infimum of $\norm{u}_J$ over such
representations.

We use the Calderon product gauge in the equivalent form
\[
\norm{f}_{\Cal}
=\inf M,
\]
where the infimum is over all measurable $b\ge |f|$ and real measurable
$s$ such that
\[
 b e^{-\theta s}\in X_0,\quad
 b e^{(1-\theta)s}\in X_1,\quad
 \norm{b e^{-\theta s}}_{X_0}\le M,\quad
 \norm{b e^{(1-\theta)s}}_{X_1}\le M .
\]
This is the usual Calderon product norm, after the standard balancing of
the two endpoint factors.

\section*{A Kernel Lemma}

\begin{lemma}[Finite logarithmic-ratio bands]
\label{lem:finite-band}
Let $g$ be a measurable function.  Suppose that for some measurable
$b\ge |g|$ and some real measurable $s$ taking values in a bounded
interval,
\[
y_0:=b e^{-\theta s}\in X_0,\qquad
y_1:=b e^{(1-\theta)s}\in X_1 .
\]
Then
\[
\norm{g}_{\PM}\le \max\{\norm{y_0}_{X_0},\norm{y_1}_{X_1}\}.
\]
\end{lemma}

\begin{proof}
First assume that $s$ takes only finitely many values.  Put
$\alpha=g/b$ on $\{b>0\}$ and $\alpha=0$ on $\{b=0\}$, so that
$|\alpha|\le 1$ and $\alpha b=g$.  Let $k_\delta$ be the normalized
indicator of $[-\delta/2,\delta/2]$.  Define
\[
u_\delta(t,\omega)=\alpha(\omega)b(\omega)k_\delta(t-s(\omega)).
\]
Because $s$ has finite range, $u_\delta$ is locally piecewise constant
as an $X_0\cap X_1$-valued function, has compact $t$-support, and is
therefore admissible in Cwikel's continuous $J$-space.  Also
\[
\int_{-\infty}^{\infty}u_\delta(t,\omega)\,dt=g(\omega).
\]
For every admissible scalar test function $\phi$ with $|\phi|\le 1$,
pointwise almost everywhere,
\[
\left|\int e^{-\theta t}\phi(t)u_\delta(t)\,dt\right|
\le
b e^{-\theta s}\int e^{-\theta r}k_\delta(r)\,dr
= C_{0,\delta}y_0,
\]
and similarly
\[
\left|\int e^{(1-\theta)t}\phi(t)u_\delta(t)\,dt\right|
\le
C_{1,\delta}y_1,
\]
where
\[
C_{0,\delta}=\int e^{-\theta r}k_\delta(r)\,dr,\qquad
C_{1,\delta}=\int e^{(1-\theta)r}k_\delta(r)\,dr .
\]
Both constants tend to $1$ as $\delta\downarrow0$.  The lattice
property gives
\[
\norm{u_\delta}_{J}
\le
\max\{C_{0,\delta}\norm{y_0}_{X_0},
      C_{1,\delta}\norm{y_1}_{X_1}\}.
\]
Letting $\delta\downarrow0$ proves the finite-valued case.

Now assume only that $s$ is bounded.  For every $\eta>0$, choose a
finite-valued measurable $q$ with $|q-s|\le \eta$.  Applying the
finite-valued case with $q$ in place of $s$ gives
\[
\norm{g}_{\PM}
\le
\max\{\norm{b e^{-\theta q}}_{X_0},
      \norm{b e^{(1-\theta)q}}_{X_1}\}
\le
e^\eta\max\{\norm{y_0}_{X_0},\norm{y_1}_{X_1}\}.
\]
Letting $\eta\downarrow0$ proves the lemma.
\end{proof}

\section*{Partial Isometry Theorem}

\begin{theorem}
\label{thm:order-continuous}
Let $X_0$ and $X_1$ be complex Banach lattices of measurable functions
on the same measure space.  Assume that both endpoint lattice norms are
order continuous.  Then, for every $0<\theta<1$,
\[
\left\langle X_0,X_1\right\rangle_{\theta,(0)}
=
\left[X_0,X_1\right]_\theta
\]
isometrically.
\end{theorem}

\begin{proof}
Cwikel proves the contractive embedding
\[
\norm{f}_{\Cplx}\le \norm{f}_{\PM}
\]
for every Banach couple \cite[Theorem 4.3]{Cwikel2015}.  It remains to
prove the reverse inequality in the present lattice setting.

Under the order-continuity hypothesis, the Calderon product
$\Cal$ gives the complex interpolation space isometrically; this is the
standard Calderon-Lozanovskii representation theorem for Banach
lattices, in the order-continuous case \cite{Calderon1964,Lozanovskii1969}.
It is therefore enough to prove
\[
\norm{f}_{\PM}\le \norm{f}_{\Cal}.
\]

Fix $M>\norm{f}_{\Cal}$.  Choose $b\ge |f|$ and real measurable $s$
such that
\[
y_0=b e^{-\theta s}\in X_0,\qquad
y_1=b e^{(1-\theta)s}\in X_1,
\]
and
\[
\norm{y_0}_{X_0}\le M,\qquad
\norm{y_1}_{X_1}\le M .
\]
For $N<K$, set
\[
f_{N,K}=f\,\one_{\{N<|s|\le K\}} .
\]
Lemma \ref{lem:finite-band}, applied to the bounded $s$-band
$\{N<|s|\le K\}$, gives
\[
\norm{f_{N,K}}_{\PM}
\le
\max\{\norm{y_0\one_{\{N<|s|\le K\}}}_{X_0},
      \norm{y_1\one_{\{N<|s|\le K\}}}_{X_1}\}.
\]
Since the endpoint norms are order continuous and the sets
$\{|s|>N\}$ decrease to the empty set, the right hand side tends to
$0$ as $N\to\infty$, uniformly in $K>N$.

Consequently the net $f\,\one_{\{|s|\le K\}}$ is Cauchy in the
continuous plus-minus norm: if $K_2>K_1$, their difference is supported
on $K_1<|s|\le K_2$ and the preceding estimate applies.  Cwikel proves
that the continuous plus-minus space is complete
\cite[Section 5]{Cwikel2015}; moreover it is continuously embedded in
$X_0+X_1$.  The same truncations converge to $f$ in $X_0+X_1$: on
$\{s>K\}$ we have $|f|\le b=e^{-(1-\theta)s}y_1\le
e^{-(1-\theta)K}y_1$, while on $\{s<-K\}$ we have
$|f|\le b=e^{\theta s}y_0\le e^{-\theta K}y_0$.  Hence the
$X_0+X_1$-norm of the discarded part is at most
$e^{-\theta K}\norm{y_0}_{X_0}+e^{-(1-\theta)K}\norm{y_1}_{X_1}$.
Therefore
\[
f=\lim_{K\to\infty} f\,\one_{\{|s|\le K\}}
\]
belongs to $\PM$.

For each fixed $K$, Lemma \ref{lem:finite-band} gives
\[
\norm{f\,\one_{\{|s|\le K\}}}_{\PM}
\le
\max\{\norm{y_0\one_{\{|s|\le K\}}}_{X_0},
      \norm{y_1\one_{\{|s|\le K\}}}_{X_1}\}
\le M .
\]
Passing to the limit in the Banach space $\PM$ yields
$\norm{f}_{\PM}\le M$.  Since $M>\norm{f}_{\Cal}$ was arbitrary,
$\norm{f}_{\PM}\le \norm{f}_{\Cal}=\norm{f}_{\Cplx}$.  Together with
Cwikel's contractive embedding, this proves the asserted isometry.
\end{proof}

\section*{Limitations}

The proof uses order continuity exactly when it removes the tails
$\{|s|>N\}$ in both endpoint lattice norms.  It does not settle Cwikel's
full conjecture for arbitrary complexified Banach lattices; in
particular, endpoint spaces with non-order-continuous norms such as
$L^\infty$-type lattices require a different argument or an obstruction.
