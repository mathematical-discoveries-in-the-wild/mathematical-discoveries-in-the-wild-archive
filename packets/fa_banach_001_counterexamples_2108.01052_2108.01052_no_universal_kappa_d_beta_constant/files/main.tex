\section*{Status and source}

This packet records a candidate full negative answer, likely valid, to
Question 1 of Gonzalez and Martinon, \emph{A quantitative approach to
disjointly non-singular operators}, arXiv:2108.01052 \cite{GM2021}.  The
question appears in Section 7.1 on PDF page 11.

\begin{center}
  \includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

The notation in the published PDF is transcribed as follows:
\begin{quote}
\textbf{Question 1.} Is $\kappa_d\leq D\cdot\beta$ for some constant
$D>0$?
\end{quote}

The answer is no.  The counterexample is an elementary strictly singular
off-diagonal perturbation of the identity on the order-continuous atomic
Banach lattice $\ell_1\oplus_\infty\ell_2$.

\section*{Definitions and two source facts}

For a Banach lattice $E$, a Banach space $Y$, and $T\in\Lin(E,Y)$, write
$\dd(E)$ for the set of disjoint sequences of nonzero vectors and
$j(A)=\inf_{\|x\|=1}\|Ax\|$.  The quantities in the question are
\[
  \tau_d(T)=\sup_{(x_n)\in\dd(E)}j\bigl(T|_{[x_n]}\bigr),
  \qquad
  \kappa_d(T)=\inf_{(x_n)\in\dd(E)}
     \tau_d\bigl(T|_{[x_n]}\bigr),
\]
and
\[
  \beta(T)=\inf\left\{
    \liminf_{n\to\infty}\|Tx_n\|:
    (x_n)\subset S_E\text{ is disjoint}
  \right\}.
\]

We also use the operational quantity
\[
  \Delta_d(R)=
  \sup_{(x_n)\in\dd(E)}
  \inf_{(y_n)\in\dd([x_n])}\left\|R|_{[y_n]}\right\|.
\]
Two results proved in the source paper are:
\begin{enumerate}
  \item $R$ is disjointly strictly singular if and only if
        $\Delta_d(R)=0$;
  \item for $A,B\in\Lin(E,Y)$,
        \[
          \kappa_d(A+B)\leq\kappa_d(A)+\Delta_d(B).
        \]
\end{enumerate}
These are Propositions 4.4 and 5.5(2), respectively, in the published
paper.  The second inequality
follows directly from $j(A+B)\leq j(A)+\|B\|$ and the two nested
disjoint-sequence operations.  In particular, adding a disjointly strictly
singular operator does not change $\kappa_d$: apply the inequality in both
directions.

\section*{Construction}

Let
\[
  E=\ell_1\oplus_\infty\ell_2,
  \qquad
  \|(x,y)\|_E=\max\{\|x\|_1,\|y\|_2\}.
\]
With the coordinatewise order, $E$ is an order-continuous atomic Banach
lattice.  Let $J:\ell_1\to\ell_2$ be the formal inclusion
$J(x_n)=(x_n)$.

\begin{lemma}\label{lem:J}
The operator $J:\ell_1\to\ell_2$ is strictly singular.
\end{lemma}

\begin{proof}
Suppose that $J$ were bounded below on an infinite-dimensional closed
subspace $M$ of $\ell_1$.  Then $J(M)$ would be a closed subspace of
$\ell_2$, and $J|_M$ would be an isomorphism from $M$ onto $J(M)$.
Consequently $M$ would be reflexive.  On the other hand, every subspace of
$\ell_1$ has the Schur property.  A reflexive Schur space is
finite-dimensional: its unit ball is weakly compact, hence weakly
sequentially compact, while the Schur property converts weakly convergent
sequences into norm-convergent sequences.  This contradiction proves the
claim.
\end{proof}

For $c>1$, define
\[
  S_c(x,y)=(0,cJx),
  \qquad
  T_c=I_E-S_c,
\]
so explicitly
\[
  T_c(x,y)=(x,y-cJx).
\]

\begin{lemma}\label{lem:S}
$S_c$ is strictly singular, $S_c^2=0$, and $T_c$ is invertible with
$T_c^{-1}=I_E+S_c$.
\end{lemma}

\begin{proof}
The nilpotence and inverse formula are immediate.  To prove strict
singularity, suppose that $S_c$ is bounded below by $\delta>0$ on an
infinite-dimensional closed subspace $M\subset E$.  For $z=(x,y)\in M$,
\[
  \delta\|z\|_E
  \leq \|S_cz\|_E
  =c\|Jx\|_2
  \leq c\|x\|_1.
\]
Thus the first-coordinate projection is bounded below on $M$, so its image
$M_1\subset\ell_1$ is a closed infinite-dimensional subspace.  Moreover,
for $x$ arising as the first coordinate of $z\in M$,
\[
  c\|Jx\|_2=\|S_cz\|_E
  \geq\delta\|z\|_E
  \geq\delta\|x\|_1.
\]
Hence $J$ is bounded below on $M_1$, contradicting Lemma~\ref{lem:J}.
\end{proof}

\section*{The full negative result}

\begin{theorem}\label{thm:counterexample}
For every $c>1$, the invertible operator $T_c\in\Lin(E)$ satisfies
\[
  \kappa_d(T_c)=1,
  \qquad
  \frac{1}{1+c}\leq\beta(T_c)\leq\frac1c.
\]
Consequently
\[
  \frac{\kappa_d(T_c)}{\beta(T_c)}\geq c,
\]
and there is no universal constant $D>0$ for which
$\kappa_d(T)\leq D\beta(T)$ holds for all Banach lattices and operators.
\end{theorem}

\begin{proof}
Since $S_c$ is strictly singular, it is disjointly strictly singular and
therefore $\Delta_d(S_c)=0$.  The perturbation inequality gives
\[
  \kappa_d(T_c)
  =\kappa_d(I_E-S_c)
  \leq\kappa_d(I_E)+\Delta_d(-S_c)=1.
\]
Applying the same inequality to $I_E=T_c+S_c$ gives
\[
  1=\kappa_d(I_E)
  \leq\kappa_d(T_c)+\Delta_d(S_c)=\kappa_d(T_c).
\]
Thus $\kappa_d(T_c)=1$.

Let $(e_n)$ denote the usual unit vectors in both $\ell_1$ and $\ell_2$,
and put
\[
  u_n=(c^{-1}e_n,e_n)\in E.
\]
The vectors $u_n$ are normalized and pairwise disjoint.  Direct computation
gives
\[
  T_cu_n=(c^{-1}e_n,0),
  \qquad
  \|T_cu_n\|_E=c^{-1}.
\]
Therefore $\beta(T_c)\leq c^{-1}$.

Finally, $T_c^{-1}=I_E+S_c$ and $\|T_c^{-1}\|\leq1+c$, so
\[
  \beta(T_c)\geq j(T_c)\geq\frac1{1+c}>0.
\]
The ratio estimate follows.  Given a proposed universal $D$, choose
$c>D$; then
\[
  D\beta(T_c)\leq D/c<1=\kappa_d(T_c),
\]
which disproves the proposed inequality.
\end{proof}

\begin{remark}
Both $\kappa_d$ and $\beta$ are homogeneous.  Thus, if a norm-one family is
desired, replace $T_c$ by $T_c/\|T_c\|$; the ratio
$\kappa_d(T_c)/\beta(T_c)$ is unchanged.  The counterexamples are not merely
DN-S: every $T_c$ is an isomorphism, and the domain is order continuous.
\end{remark}

\section*{Why the construction works}

The graph vectors $(c^{-1}e_n,e_n)$ make the off-diagonal term cancel their
entire $\ell_2$ component, so $\beta$ sees images of size $c^{-1}$.  In
contrast, $\kappa_d$ is stable under disjointly strictly singular
perturbations.  Thus the same off-diagonal term is invisible to
$\kappa_d$, even though it produces large pointwise cancellation on a
specific normalized disjoint sequence.  This is exactly the minimax gap
that a universal blocking argument would have had to eliminate.
