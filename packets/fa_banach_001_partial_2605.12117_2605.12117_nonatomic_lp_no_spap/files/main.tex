\section*{Source and Open Question}

This packet concerns Question 2 from:

\begin{quote}
Luis A. Garcia, Jose Lucas P. Luiz, Vinicius C. C. Miranda,
\emph{Norm attainment for multilinear operators and polynomials on
Banach Spaces}\\
\emph{and Banach lattices}, arXiv:2605.12117.
\end{quote}

The source location is Section 4, ``Open Questions'', Question 2. The
question appears on page 28 of the source paper:

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

The paper defines the sequentially positive approximation property
(SPAP) as follows: a Banach lattice $E$ has SPAP if there is a sequence
$(S_n)_n \subset B_{\mathcal F_+(E;E)}$ such that
\[
  \lim_{n\to\infty}\norm{|S_n-\mathrm{id}_E|(x)}=0
\]
for every $x\in E^+$. The authors note that Banach lattices with order
continuous norm whose order is given by a basis have SPAP, while
$L_2([0,1])$ and $C([0,1])$ do not.

\section*{Claim}

\begin{theorem}
Let $(\Omega,\Sigma,\mu)$ be a non-atomic finite measure space with
$0<\mu(\Omega)<\infty$, and let $1\le p<\infty$. Then $L_p(\mu)$ does
not have the SPAP.
\end{theorem}

\begin{remark}
This is a partial negative result for Question 2. It does not construct
a new Banach lattice with SPAP. It rules out the natural non-atomic
$L_p$ family, extending the paper's displayed obstruction for
$L_2([0,1])$ to all finite $p$ and all non-atomic finite measure spaces.
\end{remark}

\section*{Proof}

We first record the weakly-null sign sequence used in the obstruction.

\begin{lemma}
There is a sequence $(r_k)_{k\ge1}$ in $L_\infty(\mu)$ such that
$|r_k|=\one_\Omega$ almost everywhere for every $k$, and
$r_k\weak 0$ in $L_p(\mu)$ for every $1\le p<\infty$.
\end{lemma}

\begin{proof}
Since $\mu$ is non-atomic and finite, recursively split each set in a
finite partition into two measurable pieces of equal measure. This gives
dyadic partitions
\[
  \Omega=\bigsqcup_{\sigma\in\{0,1\}^m} A_\sigma^{(m)},\qquad
  \mu(A_\sigma^{(m)})=2^{-m}\mu(\Omega),
\]
where each $A_\sigma^{(m)}$ is split into
$A_{\sigma0}^{(m+1)}\sqcup A_{\sigma1}^{(m+1)}$. Let $\mathcal G_m$ be
the finite sigma-algebra generated by the $m$th partition, and define
\[
  r_m=\sum_{\sigma\in\{0,1\}^{m-1}}
       \big(\one_{A_{\sigma0}^{(m)}}-\one_{A_{\sigma1}^{(m)}}\big).
\]
Then $|r_m|=\one_\Omega$ and
$\mathbb E(r_m\mid\mathcal G_{m-1})=0$.

Fix $1\le p<\infty$ and let $q$ be the conjugate exponent, with
$q=\infty$ when $p=1$. For $g\in L_q(\mu)$, finite measure implies
$g\in L_1(\mu)$. Put $M_m=\mathbb E(g\mid\mathcal G_m)$. By the
martingale convergence theorem, $(M_m)_m$ converges in $L_1$ to
$\mathbb E(g\mid\bigvee_m\mathcal G_m)$, hence
$\norm{M_m-M_{m-1}}_1\to0$. Since $r_m$ is $\mathcal G_m$-measurable,
\[
  \int_\Omega g r_m\,d\mu
  =\int_\Omega M_m r_m\,d\mu
  =\int_\Omega (M_m-M_{m-1})r_m\,d\mu.
\]
The last equality uses
$\mathbb E(r_m\mid\mathcal G_{m-1})=0$. Therefore
\[
  \left|\int_\Omega g r_m\,d\mu\right|
  \le \norm{M_m-M_{m-1}}_1\longrightarrow 0.
\]
This proves $r_m\weak 0$ in $L_p(\mu)$.
\end{proof}

Assume, for contradiction, that $L_p(\mu)$ has SPAP. Then there is a
sequence $(S_n)_n\subset B_{\mathcal F_+(L_p;L_p)}$ such that
\[
  \norm{|S_n-I|(\one_\Omega)}\longrightarrow 0.
\]
Choose $n_0$ so that
\[
  \norm{|S_{n_0}-I|(\one_\Omega)}
  <\frac14\norm{\one_\Omega}_p.
\]
The operator $S_{n_0}$ has finite rank, hence it is compact. Since
$r_k\weak0$ in $L_p(\mu)$, compactness gives
\[
  \norm{S_{n_0}r_k}_p\longrightarrow0.
\]
For all sufficiently large $k$,
\[
  \norm{S_{n_0}r_k-r_k}_p
  \ge \norm{r_k}_p-\norm{S_{n_0}r_k}_p
  >\frac34\norm{\one_\Omega}_p.
\]
On the other hand, for the regular operator $S_{n_0}-I$,
\[
  |S_{n_0}-I|(|r_k|)
  \ge |(S_{n_0}-I)r_k|.
\]
Because $|r_k|=\one_\Omega$, lattice monotonicity of the $L_p$ norm gives
\[
  \norm{|S_{n_0}-I|(\one_\Omega)}_p
  \ge \norm{S_{n_0}r_k-r_k}_p
  >\frac34\norm{\one_\Omega}_p,
\]
contradicting the choice of $n_0$. Thus $L_p(\mu)$ cannot have SPAP.

\section*{Verification Notes}

The proof uses only standard Banach-lattice facts:

\begin{itemize}
\item finite-rank operators are compact;
\item compact operators send weakly null sequences to norm-null sequences;
\item for a regular operator $T$, $|T|(|x|)\ge |Tx|$;
\item the constructed dyadic signs are weakly null in finite $L_p$ spaces.
\end{itemize}

The file \texttt{code/check\_dyadic\_obstruction.py} is a finite-grid
sanity check for the conditional-expectation model behind the proof. It
is not used as evidence for the theorem.
