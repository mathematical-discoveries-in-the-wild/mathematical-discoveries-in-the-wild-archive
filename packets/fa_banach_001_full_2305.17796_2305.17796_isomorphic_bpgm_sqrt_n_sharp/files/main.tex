\section{Source question}
Koldobsky, Roysdon, and Zvavitch ask on page 5 of
\emph{Comparison Problems for Radon Transforms}~\cite{KRV} whether the
$\sqrt n$ constant in the arbitrary-measure isomorphic Busemann--Petty
inequality is optimal. Their statement is reproduced below.
\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

Let $C_n$ be the least number with the following property. For every
non-negative continuous even function $f:\R^n\to\R_+$ and origin-symmetric
convex bodies $K,L\subset\R^n$, the inequalities
\begin{equation}\label{eq:sections-source}
 \int_{K\cap\xi^\perp}f\,d\cH^{n-1}\leq
 \int_{L\cap\xi^\perp}f\,d\cH^{n-1}\quad(\xi\in S^{n-1})
\end{equation}
imply
\begin{equation}\label{eq:volume-source}
 \int_K f\,dx\leq C_n\int_L f\,dx.
\end{equation}
Koldobsky and Zvavitch proved $C_n\leq\sqrt n$~\cite{KZ}.

\section{Idea of the proof}
Klartag and Livshyts constructed a probability density whose mass inside a
convex body $A$ is bounded below, although its mass on every affine hyperplane
is only $O(n^{-1/2})$~\cite{KL}. Their explicit construction has two geometric
features crucial here: $A$ has bounded volume radius and contains an absolute
multiple of $\sqrt nB_2^n$, while the Gaussian components are centered at
distance $cn$ from the origin.

Fix a sufficiently large absolute $R$ and put $D=R\sqrt nB_2^n$. A fixed
positive amount of the slicing measure lies in $A\setminus D$. The bounded
volume radius and the inball imply, by taking two cones over any central
section of $A$, that $|A\cap H|\leq C\sqrt n|A|$. For fixed large $R$,
$D\setminus A$ therefore occupies a fixed fraction of every central section
of $D$.

Keep the slicing density only on $A\setminus D$ and place a second, tiny
continuous density on $D\setminus A$. The supports are disjoint. Scaling the
second density makes its section in every $H$ dominate the first one. Yet the
second total mass is at most
\[
 \frac{C}{\sqrt n}\frac{|D|}{|D\cap H|}=O(n^{-1/2}),
\]
whereas the first mass is bounded below. Their sum is one common continuous
even density.

\section{Formal result}
\begin{theorem}\label{thm:main}
There is a universal constant $c>0$ such that, for every $n\geq2$,
\[
 c\sqrt n\leq C_n\leq\sqrt n.
\]
Thus the $\sqrt n$ estimate in \eqref{eq:volume-source} is optimal in its
dimension dependence.
\end{theorem}
The upper bound is~\cite{KZ}; we prove the lower bound. Write
$\omega_m=|B_2^m|$.

\section{A positioned sharp slicing example}
\begin{lemma}[positioned Klartag--Livshyts example]\label{lem:KL}
There are absolute constants $a,b,c_0,M>0$ such that, for all sufficiently
large $n$, there are an origin-symmetric convex body $A\subset\R^n$ and an
even probability measure $\nu$ with continuous density $p$ satisfying
\begin{align}
 a\sqrt nB_2^n&\subset A,& |A|&\leq b^n,\label{eq:inball-volume}\\
 \sup_{\xi\in S^{n-1},t\in\R}\int_{\xi^\perp+t\xi}p\,d\cH^{n-1}
 &\leq\frac{M}{\sqrt n}.\label{eq:KL-sections}
\end{align}
Moreover, for every fixed $R>0$, after increasing the threshold on $n$,
\begin{equation}\label{eq:outside-mass}
 \nu\bigl(\operatorname{int}(A)\setminus R\sqrt nB_2^n\bigr)\geq c_0.
\end{equation}
\end{lemma}

\begin{proof}
Recall the construction in Section 4 of~\cite{KL}. Their body before the final
absolute dilation is
\[
 K_0=\operatorname{conv}\{\pm R_k\theta_{kj},\ \pm ne_i:1\leq k\leq m\},
 \qquad A=C_6K_0.
\]
Their displayed estimates (1) and (2) prove $|A|\leq b^n$ and
\eqref{eq:KL-sections} for a suitable realization. Since
$nB_1^n=\operatorname{conv}\{\pm ne_i\}\supset\sqrt nB_2^n$, the inball in
\eqref{eq:inball-volume} follows.

The measure $\nu$ is the average of standard Gaussian translates centered at
\begin{equation}\label{eq:centers}
 z=\pm\sum_{k=1}^m\widetilde R_k\theta_{kj_k},\qquad
 \widetilde R_k=\frac{n}{(\log^{(k)}n)^{3/2}}.
\end{equation}
The universal constant in the construction may harmlessly be enlarged so the
defining separation condition gives
$\widetilde R_k/\widetilde R_{k-1}\geq3$ for $2\leq k\leq m$. Also,
equation (4.1) of~\cite{KL} gives $\log^{(m)}n\leq C'_0$. Consequently
\begin{equation}\label{eq:center-lower}
 |z|\geq\widetilde R_m-\sum_{k<m}\widetilde R_k
 \geq\tfrac12\widetilde R_m\geq\frac{n}{2(C'_0)^{3/2}}=:c_1n.
\end{equation}
This is deterministic for every center.

The proof of equation (3) in~\cite{KL} observes that every center belongs to
$A/2$ and $2\sqrt nB_2^n\subset A/2$. If $G$ is standard Gaussian, then on
$\{|G|<2\sqrt n\}$ one has $z+G\in\operatorname{int}(A)$. For fixed $R$ and
large $n$, \eqref{eq:center-lower} also gives $|z+G|>R\sqrt n$. Finally,
$\mathbb P(|G|<2\sqrt n)\geq3/4$ by Markov applied to $|G|^2$. Thus one may
take $c_0=1/2$ after ignoring finitely many dimensions.
\end{proof}

\section{A ball with a uniformly thick complement}
\begin{lemma}\label{lem:thick}
If $A$ satisfies \eqref{eq:inball-volume}, there is an absolute $R>0$,
depending only on $a,b$, such that for large $n$, with
$D=R\sqrt nB_2^n$,
\begin{equation}\label{eq:thick}
 |(D\setminus A)\cap\xi^\perp|\geq\tfrac12|D\cap\xi^\perp|
 \quad(\xi\in S^{n-1}).
\end{equation}
\end{lemma}
\begin{proof}
Fix $H=\xi^\perp$. The points $\pm a\sqrt n\xi$ belong to $A$. The two cones
in $A$ with base $A\cap H$ and these apices have disjoint interiors, so
\begin{equation}\label{eq:cone}
 |A|\geq2\frac{a\sqrt n}{n}|A\cap H|,
 \qquad |A\cap H|\leq\frac{\sqrt n}{2a}b^n.
\end{equation}
There is an absolute $c>0$ with
$\omega_{n-1}\geq(c/\sqrt n)^{n-1}$. Hence
$|D\cap H|=(R\sqrt n)^{n-1}\omega_{n-1}\geq(cR)^{n-1}$.
Choose fixed $R$ so large that this is at least twice the right side of
\eqref{eq:cone} for all large $n$. Since
$|(A\cap D)\cap H|\leq|A\cap H|$, subtraction proves \eqref{eq:thick}.
\end{proof}

\section{The common-density separation construction}
\begin{lemma}\label{lem:separate}
For $A,\nu,p,D$ as above there is a non-negative continuous even density $f$
such that
\begin{equation}\label{eq:constructed-sections}
 \int_{A\cap\xi^\perp}f\,d\cH^{n-1}\leq
 \int_{D\cap\xi^\perp}f\,d\cH^{n-1}\quad(\xi\in S^{n-1}),
\end{equation}
but, for absolute $c_2,C_2>0$,
\begin{equation}\label{eq:constructed-ratio}
 \int_Af\,dx\geq c_2,\qquad \int_Df\,dx\leq\frac{C_2}{\sqrt n}.
\end{equation}
\end{lemma}
\begin{proof}
Put $U=\operatorname{int}(A)\setminus D$. By \eqref{eq:outside-mass} and inner
regularity, there is an even continuous $0\leq\chi\leq1$, vanishing outside
$U$, such that
\begin{equation}\label{eq:g-mass}
 g:=p\chi,\qquad G:=\int_{\R^n}g\geq c_0/2.
\end{equation}
As $g\leq p$, \eqref{eq:KL-sections} gives
\begin{equation}\label{eq:s}
 s:=\sup_{\xi\in S^{n-1}}\int_{\xi^\perp}g\,d\cH^{n-1}
 \leq\frac{M}{\sqrt n}.
\end{equation}

Let $V=\operatorname{int}(D)\setminus A$ and
$h_j(x)=\min\{1,j\operatorname{dist}(x,V^c)\}$. These functions are
continuous and even, vanish on $A$ and outside $D$, and increase pointwise to
$1_V$. Their central Radon transforms
$F_j(\xi)=\int_{\xi^\perp}h_j\,d\cH^{n-1}$ are continuous and increase to
\[
 F(\xi)=|V\cap\xi^\perp|=|D\cap\xi^\perp|-|(A\cap D)\cap\xi^\perp|.
\]
The limit is continuous because central-section volume varies continuously
with direction for each convex body $D$ and $A\cap D$. Dini's theorem makes
the convergence uniform. By \eqref{eq:thick}, for some $j$ the function
$h=h_j$ satisfies
\begin{equation}\label{eq:q}
 q:=\inf_{\xi\in S^{n-1}}\int_{\xi^\perp}h\,d\cH^{n-1}
 \geq\tfrac14(R\sqrt n)^{n-1}\omega_{n-1}.
\end{equation}

Set $\lambda=s/q$ and $f=g+\lambda h$. Since $g$ vanishes on $D$ and $h$ on
$A$, for every $\xi$,
\[
 \int_{A\cap\xi^\perp}f=\int_{\xi^\perp}g
 \leq s=\lambda q\leq\lambda\int_{\xi^\perp}h
 =\int_{D\cap\xi^\perp}f.
\]
Also $\int_Af=G\geq c_0/2$. Using $0\leq h\leq1$,
\eqref{eq:s}, and \eqref{eq:q},
\begin{align*}
 \int_Df&=\lambda\int_Dh\leq\frac{M}{\sqrt n}\frac{|D|}{q}\\
 &\leq\frac{4M}{\sqrt n}
 \frac{(R\sqrt n)^n\omega_n}{(R\sqrt n)^{n-1}\omega_{n-1}}
 =4MR\frac{\omega_n}{\omega_{n-1}}\leq\frac{C_2}{\sqrt n},
\end{align*}
where the last inequality is the standard gamma-ratio estimate
$\omega_n/\omega_{n-1}\leq C/\sqrt n$.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
For all large $n$, Lemma~\ref{lem:separate} gives two origin-symmetric convex
bodies $A,D$ and a common density $f$ obeying the section condition, while
\[
 \frac{\int_Af}{\int_Df}\geq c\sqrt n.
\]
Thus $C_n\geq c\sqrt n$ in all large dimensions. Also $C_n\geq1$ in every
dimension, by taking the bodies equal and a density of positive mass.
Decreasing $c$ absorbs the finitely many remaining dimensions. Combine this
with $C_n\leq\sqrt n$ from~\cite{KZ}.
\end{proof}

\section{Verification notes and limitations}
\begin{itemize}
\item The proof uses the explicit centers and body in Section 4 of~\cite{KL},
not only the abstract slicing theorem.
\item No discontinuous density is used. The cutoff on $D\setminus A$ is
uniform in direction by Dini's theorem.
\item The density is non-negative, continuous, and even. It may vanish, as the
source formulation permits.
\item The result determines the optimal order, not the exact numerical value
of $C_n$.
\item The accompanying script checks the unit-ball gamma ratio and final
$n^{-1/2}$ scaling numerically, but is not a proof dependency.
\end{itemize}

\section{Bounded novelty search}
On 21 July 2026 we searched the local 2000--2026 arXiv source corpus and the
web for the exact question and close variants, including
\texttt{isomorphic Busemann-Petty same measure sqrt n optimal} and
\texttt{arbitrary measures sharp lower bound}. The search covered the source
survey~\cite{KRV}, original upper bound~\cite{KZ}, two-measure comparison
paper~\cite{KPZ}, and sharp slicing construction~\cite{KL}. The two-measure
version has sharp $\sqrt n$ order~\cite{KPZ}, but that paper explicitly leaves
the same-measure constant open. We found no later statement of the lower bound
proved here. Novelty confidence is moderate rather than definitive.

\section{Human-review recommendation}
Check: (i) that the constants in Section 4 of~\cite{KL} may be enlarged so
consecutive $\widetilde R_k$ grow by at least $3$; (ii) the two-cone estimate
\eqref{eq:cone}; and (iii) continuity of the central-section function of
$A\cap D$, used in Dini's theorem. Once these pass, the rest is direct support
separation and scaling.
