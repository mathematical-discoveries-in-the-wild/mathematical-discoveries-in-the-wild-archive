\section*{Source and Open Question}

This packet concerns Question 8.2 from:

\begin{quote}
Ramon J. Aliaga, Luis C. Garcia-Lirola, Juan Guerrero-Viu, Matias Raja,
Abraham Rueda Zoca, \emph{Almost preserved extreme points},
arXiv:2510.21234.
\end{quote}

The paper asks for a characterization of interior almost preserved
extreme points in $\ell_p$-sums. The source location is page 37 of the
paper.

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

We use two results from the source paper. First, if $C$ is a closed,
bounded, convex subset of a Banach space $X$, then
\[
  x\in\APEP(C)
  \quad\Longleftrightarrow\quad
  x\in\overline{\ext\overline{C}^{\wstar}}^{\,\wstar}
  \subset X^{**}.
\]
Second, for $1<p<\infty$,
\[
\ext \Ball_{\left(\bigoplus_i E_i\right)_p}
=
\left\{(u_i)\in S_{\left(\bigoplus_i E_i\right)_p}:
u_i=0 \text{ or } u_i/\norm{u_i}\in \ext \Ball_{E_i}
\text{ for every }i \right\}.
\]

\section*{Notation}

Let $1<p<\infty$, let $\{X_i:i\in I\}$ be a family of Banach spaces, and
let
\[
  X=\left(\bigoplus_{i\in I}X_i\right)_p.
\]
For each $i$, put
\[
  C_i=\overline{\ext \Ball_{X_i^{**}}}^{\,\wstar}
  \subset \Ball_{X_i^{**}}.
\]
For $u\in X_i$, define the set of \emph{admissible radii}
\[
  R_i(u)=\{r\in[0,1]:u\in rC_i\},
\]
where $0C_i=\{0\}$. If $R_i(u)\neq\emptyset$, define
\[
  \rho_i(u)=\min R_i(u).
\]
If $R_i(u)=\emptyset$, set $\rho_i(u)=+\infty$.
Finally let
\[
  K=\{i\in I:X_i\neq\{0\}\}.
\]
If $K=\emptyset$, then $X=\{0\}$ and the only point of $\Ball_X$ is an
APEP. The theorem below treats the nontrivial case $K\neq\emptyset$.

\begin{lemma}
For each $i$ and $u\in X_i$, the set $R_i(u)$ is compact. In particular,
the minimum defining $\rho_i(u)$ exists whenever $R_i(u)$ is nonempty.
\end{lemma}

\begin{proof}
The set $C_i$ is weak-star compact. Suppose $r_\alpha\in R_i(u)$ and
$r_\alpha\to r$ in $[0,1]$. Choose $c_\alpha\in C_i$ with
$u=r_\alpha c_\alpha$. Passing to a subnet, $c_\alpha\to c\in C_i$ in the
weak-star topology. Then $u=rc$, so $r\in R_i(u)$.
\end{proof}

\section*{Characterization}

\begin{theorem}
Let $x=(x_i)\in\Ball_X$ with $\norm{x}<1$.

\begin{enumerate}
\item If $K$ is nonempty and finite, then $x\in\APEP(\Ball_X)$ if and
only if there exist radii $r_i\in R_i(x_i)$, $i\in K$, such that
\[
  \sum_{i\in K} r_i^p=1.
\]

\item If $K$ is infinite, then $x\in\APEP(\Ball_X)$ if and only if
\[
  \sum_{i\in I}\rho_i(x_i)^p\le 1.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
We identify
\[
  X^{**}=\left(\bigoplus_{i\in I}X_i^{**}\right)_p
  \quad\text{and}\quad
  X^*=\left(\bigoplus_{i\in I}X_i^*\right)_q,
\]
where $q$ is conjugate to $p$.

\smallskip
\noindent\emph{Necessity.}
Assume $x\in\APEP(\Ball_X)$. By the quoted characterization, there is a
net $y^\alpha=(y_i^\alpha)\in\ext\Ball_{X^{**}}$ converging weak-star to
x. Fix a finite set $F\subset I$. By compactness of $[0,1]^F$, pass to a
subnet so that
\[
  \norm{y_i^\alpha}\to r_i
  \quad (i\in F).
\]
If $r_i=0$, then $x_i=0$, so $r_i\in R_i(x_i)$. If $r_i>0$, then
eventually $y_i^\alpha\neq0$, and
\[
  e_i^\alpha=\frac{y_i^\alpha}{\norm{y_i^\alpha}}
  \in\ext\Ball_{X_i^{**}}.
\]
Passing to a further subnet, $e_i^\alpha\to e_i\in C_i$ weak-star, and
$x_i=r_i e_i$. Thus $r_i\in R_i(x_i)$ and
$\rho_i(x_i)\le r_i$.
Since each $y^\alpha$ has norm one,
\[
  \sum_{i\in F}\rho_i(x_i)^p
  \le
  \sum_{i\in F}r_i^p
  =
  \lim_\alpha\sum_{i\in F}\norm{y_i^\alpha}^p
  \le 1.
\]
Taking the supremum over finite $F$ gives
\[
  \sum_{i\in I}\rho_i(x_i)^p\le 1.
\]
This proves the necessity in the infinite case.

If $K$ is finite, apply the same argument with $F=K$. Since all nonzero
summands are indexed by $K$, the limiting radii satisfy
\[
  \sum_{i\in K}r_i^p
  =
  \lim_\alpha \sum_{i\in K}\norm{y_i^\alpha}^p
  =1,
\]
and each $r_i$ belongs to $R_i(x_i)$.

\smallskip
\noindent\emph{Sufficiency when $K$ is finite.}
Assume there are $r_i\in R_i(x_i)$ with $\sum_{i\in K}r_i^p=1$.
Let
\[
  U=\{z\in\Ball_{X^{**}}:|f^k(z-x)|<\varepsilon,\ k=1,\ldots,m\}
\]
be a basic weak-star neighborhood of $x$.
For every $i\in K$ with $r_i>0$, choose $c_i\in C_i$ with $x_i=r_i c_i$.
Since $c_i$ belongs to the weak-star closure of $\ext\Ball_{X_i^{**}}$,
we may choose $e_i\in\ext\Ball_{X_i^{**}}$ so close to $c_i$ on the
finitely many coordinate functionals $f_i^1,\ldots,f_i^m$ that the vector
$y$ defined by
\[
  y_i=
  \begin{cases}
    r_i e_i, & r_i>0,\\
    0, & r_i=0
  \end{cases}
\]
satisfies $|f^k(y-x)|<\varepsilon$ for all $k$.
The equality $\sum_{i\in K}r_i^p=1$ and the extreme-point formula show
that $y\in\ext\Ball_{X^{**}}$. Hence every weak-star neighborhood of
$x$ meets $\ext\Ball_{X^{**}}$, so $x\in\APEP(\Ball_X)$.

\smallskip
\noindent\emph{Sufficiency when $K$ is infinite.}
Assume $\sum_i\rho_i(x_i)^p\le1$. Let $U$ be a basic weak-star
neighborhood of $x$ as above. Because $q<\infty$, finite-support
functionals are norm dense in $X^*=(\bigoplus_i X_i^*)_q$. Choose
finite-support $g^1,\ldots,g^m\in X^*$, supported on a common finite set
F, such that
\[
  \norm{f^k-g^k}<\varepsilon/4
  \quad (k=1,\ldots,m).
\]
Put
\[
  s=\sum_{i\in F}\rho_i(x_i)^p.
\]
Then $s\le1$.

For each $i\in F$ with $\rho_i(x_i)>0$, choose
$c_i\in C_i$ such that $x_i=\rho_i(x_i)c_i$, and then choose
$e_i\in\ext\Ball_{X_i^{**}}$ close enough to $c_i$ on the finitely many
coordinate functionals $g_i^1,\ldots,g_i^m$ that
\[
 \sum_{i\in F}
 |g_i^k(\rho_i(x_i)e_i-x_i)|<\varepsilon/2
 \quad (k=1,\ldots,m).
\]
If $\rho_i(x_i)=0$, set the corresponding coordinate equal to zero.

If $s<1$, choose $i_0\in K\setminus F$ and choose an extreme point
$e_{i_0}\in\ext\Ball_{X_{i_0}^{**}}$; this is possible by Krein-Milman
because $X_{i_0}\neq\{0\}$. Put $r_0=(1-s)^{1/p}$. Define $y\in X^{**}$
by
\[
 y_i=
 \begin{cases}
   \rho_i(x_i)e_i, & i\in F,\ \rho_i(x_i)>0,\\
   r_0e_{i_0}, & i=i_0,\ s<1,\\
   0, & \text{otherwise.}
 \end{cases}
\]
If $s=1$, use the same definition but omit the $i_0$ line.
In both cases $\norm{y}=1$, and every nonzero normalized coordinate is an
extreme point of the corresponding coordinate ball. Hence
$y\in\ext\Ball_{X^{**}}$.

For each $k$,
\[
  |g^k(y-x)|
  =
  \left|\sum_{i\in F}g_i^k(y_i-x_i)\right|
  <\varepsilon/2.
\]
Also $\norm{y-x}\le2$, so
\[
  |(f^k-g^k)(y-x)|
  \le \norm{f^k-g^k}\,\norm{y-x}
  <\varepsilon/2.
\]
Thus $|f^k(y-x)|<\varepsilon$ for every $k$, and
$y\in U\cap\ext\Ball_{X^{**}}$. The weak-star closure characterization
therefore gives $x\in\APEP(\Ball_X)$.
\end{proof}

\section*{Checks and Consequences}

If $\norm{x}=1$, then $\sum_i\norm{x_i}^p=1$ and
$\rho_i(x_i)\ge\norm{x_i}$. The theorem therefore forces
$\rho_i(x_i)=\norm{x_i}$ for all nonzero coordinates, which is exactly
the source paper's norm-one characterization: each nonzero normalized
coordinate must be an APEP in its coordinate ball.

The earlier sufficient condition in Proposition 8.3 of the source paper
is recovered by taking the original coordinate radii on the nonzero
support and using an admissible positive radius at a zero coordinate
where $0$ is APEP. The infinite-summand partial result from the previous
packet is the special case where
$\rho_i(x_i)=\norm{x_i}$ for every nonzero coordinate and the leftover
mass is put outside the finite weak-star test.

\section*{Verification Notes}

No computational verifier was used. The argument is structural and rests
on weak-star compactness, finite-support approximation in
$(\bigoplus_i X_i^*)_q$, and the extreme-point formula for $\ell_p$-sums.

Human review should focus on the distinction between the finite and
infinite nonzero-summand cases: finite sums require admissible radii whose
$p$-sum is exactly one, while infinite sums require only that the least
admissible radii have $p$-sum at most one.

\section*{Limitations and Novelty Check}

This is a candidate full answer to Question 8.2 as stated. It is an
abstract characterization rather than a closed-form list for every
coordinate space; the coordinate information is encoded in the weak-star
closures $C_i=\overline{\ext\Ball_{X_i^{**}}}^{\,\wstar}$, which are
already central in the source paper's APEP characterization.
