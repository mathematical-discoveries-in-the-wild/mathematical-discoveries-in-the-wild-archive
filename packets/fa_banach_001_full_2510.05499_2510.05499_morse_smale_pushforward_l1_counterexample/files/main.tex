\begin{abstract}
We disprove Conjecture~1 of Tikhomirov's arXiv:2510.05499.  There is a smooth
Morse--Smale diffeomorphism $h$ of a closed surface for which the pushforward
operator $S_h$ on $L^1(M,TM)$ is not generalized hyperbolic.  The obstruction
comes from the volume Jacobian in the $L^1$ norm.  A saddle with derivative
$\operatorname{diag}(1/2,4)$ has a Jacobian-neutral stable covector direction;
long local orbit towers then show that $S_h^*-I$ is not bounded below.  Hence
$S_h-I$ is not onto, whereas a Green-series argument shows that $I-T$ is onto
for every generalized hyperbolic isomorphism $T$.
\end{abstract}

\section{The conjecture}

For a diffeomorphism $h\colon M\to M$, the source defines the pushforward on
vector fields by
\[
  (S_hv)(h(x))=Dh(x)v(x).
\]
It asks whether Axiom A and the Strong Transversality Condition force $S_h$
to be generalized hyperbolic on every $L^p(M,TM)$, $1\leq p\leq\infty$, and
on bounded continuous vector fields \cite[Conjecture~1]{Tikhomirov}:

\begin{center}
\includegraphics[width=.96\textwidth]{figures/conjecture_crop.png}
\end{center}

We use $L^1$ with respect to a smooth Riemannian volume.  This is one of the
spaces explicitly included in the conjecture.

\begin{theorem}\label{thm:counterexample}
There are a closed smooth surface $M$ and a $C^\infty$ Morse--Smale
diffeomorphism $h\colon M\to M$ such that the bounded isomorphism
\[
 S_h\colon L^1(M,TM)\longrightarrow L^1(M,TM)
\]
is not generalized hyperbolic.  In particular, Conjecture~1 of
\cite{Tikhomirov} is false.
\end{theorem}

The proof has two short ingredients.

\section{A necessary consequence of generalized hyperbolicity}

Recall that an isomorphism $T$ of a Banach space is generalized hyperbolic if
there is a bounded splitting $X=E^s\oplus E^u$ such that
$TE^s\subset E^s$, $T^{-1}E^u\subset E^u$, and the corresponding forward and
backward powers contract exponentially.

\begin{lemma}[Green right inverse]\label{lem:green}
If $T$ is generalized hyperbolic, then $I-T$ is surjective.  In fact, if
$P,Q$ are the projections onto $E^s,E^u$, respectively, then
\begin{equation}
 Gy=\sum_{n=0}^{\infty}T^nPy-\sum_{n=1}^{\infty}T^{-n}Qy
 \label{eq:green}
\end{equation}
defines a bounded linear operator satisfying $(I-T)G=I$.
\end{lemma}

\begin{proof}
The two series converge in norm by the exponential estimates and the
inclusions $TE^s\subset E^s$, $T^{-1}E^u\subset E^u$.  Their partial sums
give, by telescoping,
\begin{align*}
 (I-T)\sum_{n=0}^{\infty}T^nPy&=Py,\\
 (I-T)\left(-\sum_{n=1}^{\infty}T^{-n}Qy\right)&=Qy.
\end{align*}
Since $P+Q=I$, the claim follows.
\end{proof}

We shall also use the standard dual consequence of the open mapping theorem:
if a bounded operator $A\colon X\to Y$ is onto, then there is $c>0$ such that
\begin{equation}
 \|A^*y^*\|\geq c\|y^*\|\qquad(y^*\in Y^*).
 \label{eq:adjoint-below}
\end{equation}
Indeed, openness gives a $K$ such that every unit vector of $Y$ has a
preimage of norm at most $K$, and then
$\|y^*\|\leq K\|A^*y^*\|$.

\section{A Jacobian-neutral Morse--Smale saddle}

We first record the elementary geometric realization used below.

\begin{lemma}\label{lem:realization}
There is a Morse--Smale gradient flow on a closed surface with a saddle $o$
and a coordinate neighborhood in which its vector field is
\[
  \dot x=-(\log2)x,\qquad \dot y=(\log4)y.
\]
The Riemannian metric and volume can be chosen Euclidean in this neighborhood.
Consequently, its time-one map is locally
\begin{equation}
  h(x,y)=A(x,y),\qquad
  A=\begin{pmatrix}1/2&0\\0&4\end{pmatrix}.
  \label{eq:localA}
\end{equation}
It is an Axiom-A diffeomorphism satisfying the Strong Transversality
Condition.
\end{lemma}

\begin{proof}
Take a Morse function on $S^2$ with two minima, one saddle and one maximum.
In a Morse chart at the saddle prescribe the displayed quadratic negative
gradient field and a Euclidean metric.  A standard relative Morse--Smale
perturbation, supported away from the critical-point charts, makes all stable
and unstable manifolds transverse without changing this local model.  The
nonwandering set of the time-one map consists of the finitely many hyperbolic
equilibria, so the map satisfies Axiom A; the Morse--Smale transversality is
exactly the Strong Transversality Condition for these invariant manifolds.
The time-one map of the displayed local linear field is \eqref{eq:localA}.
\end{proof}

Fix such an $h$ and write $S=S_h$ on $L^1(M,TM)$.  Compactness of $M$ and
smoothness of $h,h^{-1}$ make $S$ a bounded linear isomorphism.  Its Banach
adjoint acts on essentially bounded covector fields.  Change of variables in
the natural pairing gives
\begin{equation}
  (S^*\psi)(x)=J_h(x)\,Dh(x)^*\psi(h(x)),
  \label{eq:adjoint}
\end{equation}
where $J_h$ is the volume Jacobian.  In the local chart of
Lemma~\ref{lem:realization}, let $e^1$ be the first coordinate covector.  Since
\[
 J_h=\det A=2,\qquad A^*e^1=\tfrac12e^1,
\]
formula \eqref{eq:adjoint} becomes simply
\begin{equation}
 S^*(g e^1)(x)=g(Ax)e^1
 \label{eq:neutral}
\end{equation}
whenever $x$ and $Ax$ lie in the linear chart.  This exact cancellation is
the obstruction.

\begin{lemma}[Long adjoint towers]\label{lem:towers}
There are $\psi_N\in L^\infty(M,T^*M)$, $N\geq2$, such that
\begin{equation}
 \|\psi_N\|_\infty=1,
 \qquad
 \|(S^*-I)\psi_N\|_\infty\leq\frac1N.
 \label{eq:tower-estimate}
\end{equation}
\end{lemma}

\begin{proof}
For each $N$, choose a nonzero point $z_N$ sufficiently close to the saddle
that the finite orbit segment
\[
 \{A^kz_N:-N-2\leq k\leq N+2\}
\]
lies in the linear chart.  For example, in a fixed small coordinate rectangle
one may take the two coordinates of $z_N$ of orders $2^{-(N+2)}$ and
$4^{-(N+2)}$.  The points of this finite segment are distinct.  Hence there
is a positive-measure neighborhood $E_N$ of $z_N$ whose images
\[
 E_{k,N}=h^k(E_N)=A^k(E_N),\qquad -N-1\leq k\leq N+1,
\]
are pairwise disjoint and stay in the linear chart.

Extend
\[
 c_{k,N}=\max\left\{1-\frac{|k|}{N},0\right\}
\]
to all $k\in\mathbb Z$, and define the measurable covector field
\[
 \psi_N(x)=
 \begin{cases}
   c_{k,N}e^1,&x\in E_{k,N},\ -N\leq k\leq N,\\
   0,&\text{otherwise}.
 \end{cases}
\]
The tower levels are disjoint and $c_{0,N}=1$, so
$\|\psi_N\|_\infty=1$.  By \eqref{eq:neutral}, on the predecessor of the
$k$th level the coefficient of $S^*\psi_N$ is $c_{k,N}$.  Thus every
coefficient of $(S^*-I)\psi_N$ is a difference of two adjacent $c_{k,N}$'s.
The coefficients vanish at and beyond both ends of the tower, and
\[
 \sup_{k\in\mathbb Z}|c_{k+1,N}-c_{k,N}|=\frac1N.
\]
This proves \eqref{eq:tower-estimate}.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:counterexample}]
Lemma~\ref{lem:towers} shows that $S^*-I$ is not bounded below.  By
\eqref{eq:adjoint-below}, $S-I$ is therefore not surjective.  If $S$ were
generalized hyperbolic, Lemma~\ref{lem:green} would say that $I-S$ is
surjective, a contradiction.  The diffeomorphism $h$ has Axiom A and strong
transversality by Lemma~\ref{lem:realization}, and $p=1$ is included in the
universal claim of Conjecture~1.
\end{proof}

\section{Remarks on scope and novelty}

The counterexample uses the interaction between the derivative cocycle and
the volume change in an $L^1$ norm.  It does not decide the conjectured
conclusion for volume-preserving Axiom-A maps, for $C^0$ vector fields, or for
other fixed values of $p$ under additional bunching assumptions.

Weighted operators generated by Morse--Smale maps have an established
spectral theory \cite{AntonevichMakowska}, and generalized hyperbolicity has
been characterized for significant classes of $L^p$ operators
\cite{DAnielloDarjiMaiuriello}.  A bounded search on 21 July 2026 found no
statement of the local Jacobian-neutral tower obstruction above.  This is a
novelty check, not a definitive certification.
