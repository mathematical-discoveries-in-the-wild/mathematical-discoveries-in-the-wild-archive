\section*{Source Question}

The source paper is Ondrej F. K. Kalenda,
\emph{Quantitative Schur property and measures of weak non-compactness},
arXiv:2505.12893 \cite{kalenda2025}.  Its final Question~9.2 asks whether
the implication
\[
  (iii_c)\Longrightarrow (ii_{2c+1})
\]
in Proposition~\textup{P:schur} is optimal.  The same proposition proves
\((i_C)\iff(ii_C)\), so this is equivalently a question about the best
constant \(C\) in \((iii_c)\Rightarrow(i_C)\).

\section*{Definitions}

For a bounded sequence \((x_n)\) in a Banach space \(X\), write
\[
  \delta(x_n)=\diam \clu(x_n)\subset X^{**}
\]
and
\[
  \wca(x_n)=\inf\{\ca(x_{n_j}) : n_1<n_2<\cdots\}.
\]
Condition \((iii_c)\) in Proposition~\textup{P:schur} is
\[
  \wca(x_n)\le c\,\delta(x_n)
  \qquad\text{for every bounded sequence }(x_n)\subset X.
\]
Condition \((i_C)\) is
\[
  \limsup_n \|x_n\|\le
  C\,\hdist(\clu(x_n),\{0\})
  \qquad\text{for every bounded sequence }(x_n)\subset X.
\]
Here \(\hdist(\clu(x_n),\{0\})=\sup\{\|x^{**}\|:x^{**}\in\clu(x_n)\}\).

\section*{Answer}

\begin{theorem}\label{thm:answer}
The best constant forced by \((iii_c)\) in Proposition~\textup{P:schur} is
\[
  \begin{cases}
    1, & 0<c<1,\\
    2c+1, & c\ge 1.
  \end{cases}
\]
More precisely, if \(0<c<1\) and \(X\) satisfies \((iii_c)\), then \(X\) is
finite-dimensional and hence satisfies \((ii_1)\).  The constant \(1\) is
sharp on nonzero finite-dimensional spaces.  If \(c\ge1\), there is a Banach
space satisfying \((iii_c)\) and failing \((i_C)\), equivalently \((ii_C)\),
for every \(C<2c+1\).  Thus Kalenda's implication
\((iii_c)\Rightarrow(ii_{2c+1})\) is sharp throughout the nontrivial
infinite-dimensional range \(c\ge1\).
\end{theorem}

\section*{The sharp examples for \(c\ge1\)}

Fix \(c\ge1\), and let \(\K\) be either \(\mathbb R\) or \(\mathbb C\).  On
\(\K\oplus\ellone\) define
\[
  \|(t,x)\|_c
  =
  \max\bigl\{\|x\|_1,\ |t|+2c\|x\|_\infty\bigr\}.
\]
This is an equivalent norm on the direct sum; call the resulting Banach space
\(X_c\).

\begin{lemma}[a gliding-hump estimate]\label{lem:gliding}
Let \((u_n)\) be a bounded sequence in \(\ellone\) which converges
coordinatewise to \(0\), and suppose that \(\|u_n\|_1\to A\).  Then:
\begin{enumerate}
\item \(\delta_{\ellone}(u_n)\ge 2A\);
\item for every \(\varepsilon>0\) there is a subsequence, still denoted
      \((u_n)\), such that for all \(p<q\),
      \[
        \|u_p-u_q\|_1\le 2A+\varepsilon,
        \qquad
        \|u_p-u_q\|_\infty\le A+\varepsilon .
      \]
\end{enumerate}
\end{lemma}

\begin{proof}
For the first assertion, fix \(\varepsilon>0\).  A standard gliding-hump
selection gives a subsequence \((v_m)\) of \((u_n)\) such that
\[
  \|v_m-v_n\|_1\ge 2A-\varepsilon\qquad(m\ne n).
\]
Indeed, when finitely many earlier vectors have been chosen, choose finite
sets carrying almost all of their \(\ellone\)-mass and then use coordinatewise
convergence to make the next vector small on those sets.  Hence
\(\ca(v_m)\ge2A-\varepsilon\).  Since \(\ellone\) has the \(1\)-Schur
property, \(\ca(v_m)\le\delta_{\ellone}(v_m)\), and cluster points of a
subsequence are cluster points of the original sequence.  Therefore
\(\delta_{\ellone}(u_n)\ge2A-\varepsilon\), and \(\varepsilon\downarrow0\)
proves the claim.

For the second assertion, discard finitely many terms so that
\(\|u_n\|_1\le A+\varepsilon/4\) for all remaining \(n\).  Choose a subsequence
inductively as follows.  After \(u_{n_1},\dots,u_{n_m}\) have been chosen, pick
a finite set \(F_m\subset\N\) with
\[
  \sum_{j\notin F_m}|u_{n_m}(j)|<\varepsilon/4.
\]
Then choose all later indices so large that their coordinates on
\(\bigcup_{r\le m}F_r\) have modulus \(<\varepsilon/4\).  Relabel this
subsequence as \((u_n)\).  The \(\ellone\)-estimate is immediate from the
triangle inequality.  For \(p<q\), on \(F_p\) the vector \(u_q\) is uniformly
small, while off \(F_p\) the vector \(u_p\) is uniformly small.  Since
\(\|u_p\|_\infty,\|u_q\|_\infty\le A+\varepsilon/4\), this gives
\(\|u_p-u_q\|_\infty\le A+\varepsilon\).
\end{proof}

\begin{proposition}\label{prop:iii}
For every \(c\ge1\), the space \(X_c\) satisfies \((iii_c)\).
\end{proposition}

\begin{proof}
Let \(z_n=(t_n,x_n)\) be a bounded sequence in \(X_c\).  To estimate
\(\wca(z_n)\), pass to a subsequence such that \(t_n\to t\), \(x_n\) converges
coordinatewise to some \(x\in\ellone\), and
\[
  \|x_n-x\|_1\longrightarrow A.
\]
The existence of \(x\in\ellone\) follows from Fatou's lemma applied to the
coordinatewise limit of the \(\ellone\)-bounded sequence \((x_n)\).

Set \(u_n=x_n-x\).  The projection \(P:X_c\to\ellone\), \(P(t,x)=x\), has
norm at most \(1\).  Therefore
\[
  \delta_{X_c}(z_n)\ge \delta_{\ellone}(x_n)
  =\delta_{\ellone}(u_n)\ge 2A
\]
by Lemma~\ref{lem:gliding}: indeed, every weak-star cluster point of
\((x_n)\) is the \(P^{**}\)-image of a weak-star cluster point of a subnet of
\((z_n)\), and \(P^{**}\) is a contraction.  Now fix \(\varepsilon>0\).  Passing to a further
subsequence and using Lemma~\ref{lem:gliding}, we may suppose that
\[
  |t_p-t_q|\le\varepsilon,\qquad
  \|u_p-u_q\|_1\le2A+\varepsilon,\qquad
  \|u_p-u_q\|_\infty\le A+\varepsilon
\]
for all \(p<q\).  Hence
\[
\begin{aligned}
  \|z_p-z_q\|_c
  &=
  \max\{\|u_p-u_q\|_1,\ |t_p-t_q|+2c\|u_p-u_q\|_\infty\} \\
  &\le
  \max\{2A+\varepsilon,\ \varepsilon+2c(A+\varepsilon)\}.
\end{aligned}
\]
Since \(c\ge1\), the right hand side is at most
\(2cA+(2c+1)\varepsilon\).  Thus this further subsequence has Cauchy
amplitude at most \(2cA+(2c+1)\varepsilon\).  Letting
\(\varepsilon\downarrow0\) gives
\[
  \wca(z_n)\le 2cA\le c\,\delta_{X_c}(z_n).
\]
Because the original sequence was arbitrary, \(X_c\) satisfies \((iii_c)\).
\end{proof}

\begin{proposition}\label{prop:cluster}
The sequence \(y_n=(1,e_n)\) in \(X_c\) satisfies
\[
  \|y_n\|_c=2c+1
\]
for every \(n\), while every weak-star cluster point of \((y_n)\) in
\(X_c^{**}\) has norm exactly \(1\).
\end{proposition}

\begin{proof}
The norm equality is immediate:
\[
  \|(1,e_n)\|_c=\max\{1,1+2c\}=2c+1.
\]
Let \(y^{**}\in\clu(y_n)\).  The functional \(\pi(t,x)=t\) has norm at most
\(1\) on \(X_c\), so \(\|y^{**}\|\ge |y^{**}(\pi)|=1\).

We prove the reverse inequality.  Let \(f\in X_c^*\) with \(\|f\|\le1\).  Write
\[
  f(t,x)=s\,t+a(x),
\]
where \(s\in\K\) and \(a\in\ell_\infty=(\ellone)^*\).  Suppose, toward a
contradiction, that \(|y^{**}(f)|>1\).  Multiplying \(f\) by a scalar of
modulus one and then passing to infinitely many indices from the cluster
subnet, we may find \(\eta>0\) and distinct indices \(n_1,n_2,\dots\) such
that
\[
  \operatorname{Re} f(1,e_{n_j})>1+\eta
  \qquad(j\in\N).
\]
For \(m>2c\), put
\[
  v_m=\left(1-\frac{2c}{m},\ \frac1m\sum_{j=1}^m e_{n_j}\right).
\]
Then
\[
  \|v_m\|_c
  =
  \max\left\{1,\ 1-\frac{2c}{m}+\frac{2c}{m}\right\}
  =1.
\]
Also \(|s|\le1\), because \(\|(1,0)\|_c=1\).  Therefore
\[
\begin{aligned}
  \operatorname{Re} f(v_m)
  &=
  \frac1m\sum_{j=1}^m \operatorname{Re} f(1,e_{n_j})
   -\frac{2c}{m}\operatorname{Re}s  \\
  &> 1+\eta-\frac{2c}{m}.
\end{aligned}
\]
For \(m\) large this is \(>1\), contradicting \(\|f\|\le1\) and
\(\|v_m\|_c=1\).  Hence \(|y^{**}(f)|\le1\) for all \(\|f\|\le1\), so
\(\|y^{**}\|\le1\).
\end{proof}

\begin{proof}[Proof of sharpness for \(c\ge1\)]
By Propositions~\ref{prop:iii} and~\ref{prop:cluster}, \(X_c\) satisfies
\((iii_c)\), while for \(y_n=(1,e_n)\)
\[
  \limsup_n\|y_n\|_c=2c+1
  \quad\text{and}\quad
  \hdist(\clu(y_n),\{0\})=1.
\]
Thus \((i_C)\) fails for every \(C<2c+1\).  Since Proposition~\textup{P:schur}
in the source paper proves \((i_C)\iff(ii_C)\), \((ii_C)\) also fails for
every \(C<2c+1\).  The source proposition proves the matching upper
implication \((iii_c)\Rightarrow(ii_{2c+1})\), so \(2c+1\) is sharp for
\(c\ge1\).
\end{proof}

\section*{The small-constant range \(0<c<1\)}

\begin{proposition}\label{prop:smallc}
If \(0<c<1\) and \(X\) satisfies \((iii_c)\), then \(X\) is finite-dimensional.
Consequently \(X\) satisfies \((ii_1)\).
\end{proposition}

\begin{proof}
The proof of Proposition~\textup{P:schur} in the source paper shows that
\((iii_c)\) implies the Schur property.  Suppose that \(X\) were
infinite-dimensional.  By Rosenthal's \(\ell_1\) theorem, together with the
Schur property's weak sequential completeness, \(X\) contains an isomorphic
copy of \(\ell_1\).  James' distortion theorem for \(\ell_1\) then says that,
for every \(\varepsilon>0\), \(X\) contains a normalized sequence
\((x_n)\) which is \((1-\varepsilon)^{-1}\)-equivalent to the unit vector
basis of \(\ell_1\); in particular
\[
  \left\|\sum_{j=1}^m \alpha_j x_j\right\|
  \ge (1-\varepsilon)\sum_{j=1}^m|\alpha_j|
\]
for all finitely supported scalar families \((\alpha_j)\).
Then every subsequence of \((x_n)\) has Cauchy amplitude at least
\(2(1-\varepsilon)\), while \(\delta(x_n)\le2\), since \(\|x_n\|=1\).  Applying
\((iii_c)\) to \((x_n)\) gives
\[
  2(1-\varepsilon)\le \wca(x_n)\le c\,\delta(x_n)\le2c.
\]
Letting \(\varepsilon\downarrow0\) yields \(c\ge1\), a contradiction.

Thus \(X\) is finite-dimensional.  Every bounded sequence in a
finite-dimensional space has a norm-convergent subsequence, so
\(\wca(x_n)=0\) for all bounded sequences.  Moreover compactness gives
\(\ca(x_n)\le\delta(x_n)\), which is exactly \((ii_1)\).
\end{proof}

\begin{proof}[Sharpness for \(0<c<1\)]
Let \(X\) be any nonzero finite-dimensional Banach space.  As above, \(X\)
satisfies \((iii_c)\) for every \(c>0\).  Choose \(x\in X\setminus\{0\}\) and
consider the alternating sequence \(x,-x,x,-x,\dots\).  Its Cauchy amplitude
is \(2\|x\|\), and its weak-star cluster set is \(\{x,-x\}\), which has
diameter \(2\|x\|\).  Hence \((ii_C)\) fails for every \(C<1\).  Therefore
the constant \(1\) in Proposition~\ref{prop:smallc} is best possible.
\end{proof}

\section*{Verification and novelty notes}

\(2c+1\), and the averaging vectors used in Proposition~\ref{prop:cluster}
