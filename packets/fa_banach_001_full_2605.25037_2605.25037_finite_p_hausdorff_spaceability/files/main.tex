\section*{Source and Target}

Source paper: Jia Liu, Saisai Shi, and Zhenliang Zhang,
\emph{On strong spaceability of continuous functions and fractal dimensions},
arXiv:2605.25037v2.

The relevant open problem is Question 3.19 on page 24 of the source paper.
The packet includes the source PDF as \texttt{source\_paper.pdf}; the following
crop is rendered from that PDF.

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

For \(1<s\leq 2\), put
\[
H_s[0,1]=\{f\in C[0,1]:\dimH G_f([0,1])=s\}.
\]
Question 3.19 asks whether \(H_s[0,1]\) is \((p,\cc)\)-spaceable for every
integer \(3\leq p<\aleph_0\).

\begin{definition}
Let \(X\) be a topological vector space, \(A\subset X\), and \(p\in\NN\).
The set \(A\) is \((p,\cc)\)-spaceable if, whenever \(E\) is a \(p\)-dimensional
linear subspace of \(X\) with
\[
E\subset A\cup\{0\},
\]
there is a closed linear subspace \(Y\subset X\) such that
\[
E\subset Y\subset A\cup\{0\}
\quad\text{and}\quad
\dim Y=\cc.
\]
\end{definition}

\section*{Imported Result}

For a set \(K\subset[0,1]\), write \(u\in H_s(K)\) to mean
\(\dimH G_u(K)=s\).

\begin{proposition}[Liu--Shi--Zhang, Proposition 3.10 with interval domain and \(m=1\)]
Let \(1<s\leq2\), let \(p\in\NN\), and let
\(f_1,\ldots,f_p\in H_s[0,1]\) be linearly independent.  Assume
\[
\spanop\{f_1,\ldots,f_p\}\subset H_s[0,1]\cup\{0\}.
\]
Then there is a compact set \(K_0\subset[0,1]\) such that
\[
\dimH K_0=s-1
\]
and, for every nonzero \((a_1,\ldots,a_p)\in\RR^p\),
\[
\left.\sum_{i=1}^p a_i f_i\right|_{[0,1]\setminus K_0}
\in H_s([0,1]\setminus K_0).
\]
\end{proposition}

\section*{Auxiliary Facts}

We use the following standard facts.

\begin{enumerate}[label=(F\arabic*),leftmargin=*]
\item Hausdorff dimension is countably stable:
\[
\dimH\left(\bigcup_{n=1}^{\infty} A_n\right)=\sup_n \dimH A_n.
\]
\item If \(K\subset\RR\) and \(f\in C(K)\), then
\[
\dimH G_f(K)\leq \dimH K+1.
\]
Indeed, \(G_f(K)\subset K\times f(K)\) and \(f(K)\subset\RR\).
\item If \(K\subset[0,1]\) is compact and \(d=\dimH K\), then \(K\) has a
full-dimensional point: there is \(x_0\in K\) such that
\[
\dimH(K\cap(x_0-r,x_0+r))=d,\qquad r>0.
\]
Otherwise finitely many relatively open neighborhoods of dimension \(<d\)
would cover \(K\), contradicting finite stability.
\item If \(E\subset\RR\) is an uncountable compact set, then there exists
\(u\in C(E)\) such that
\[
\dimH G_u(E)=\dimH E+1.
\]
This is the prevalent maximal graph-dimension theorem cited as Lemma 2.8 in
the source paper.
\end{enumerate}

\section*{A Compact \(c_0\)-Subspace Lemma}

\begin{lemma}\label{lem:c0-compact}
Let \(K\subset[0,1]\) be compact and let \(d=\dimH K>0\).  There is a closed
subspace \(X_K\subset C[0,1]\), isomorphic to \(c_0\), with the following
properties.
\begin{enumerate}[label=(\roman*),leftmargin=*]
\item For every nonzero \(h\in X_K\),
\[
\dimH G_h(K)=d+1.
\]
\item Every \(h\in X_K\) is affine on each connected component of
\([0,1]\setminus K\), with constant extension on endpoint components if they
occur.
\end{enumerate}
\end{lemma}

\begin{proof}
Choose a full-dimensional point \(x_0\in K\).  Let \(r_m\downarrow0\), and put
\[
A_m=K\cap\bigl((x_0-r_m,x_0+r_m)\setminus [x_0-r_{m+1},x_0+r_{m+1}]\bigr).
\]
The annuli are pairwise disjoint.  Since
\[
K\cap(x_0-r_N,x_0+r_N)\subset \{x_0\}\cup\bigcup_{m\geq N} A_m
\]
has Hausdorff dimension \(d\), countable stability gives
\[
\sup_{m\geq N}\dimH A_m=d,\qquad N\in\NN.
\]
Passing to a subsequence and using inner regularity of Hausdorff dimension,
choose compact sets \(E_n\subset K\) and open intervals \(I_n\) such that
\[
E_n\subset I_n,\qquad
\overline{I_n}\cap\overline{I_m}=\emptyset\ (n\neq m),\qquad
I_n\to x_0,
\]
and
\[
d_n:=\dimH E_n>d-\frac1n.
\]
Discarding finitely many terms, assume \(d_n>0\) for all \(n\).  Then
\(d_n\to d\), and each \(E_n\) is uncountable.

By (F4), choose \(\varphi_n\in C(E_n)\) with
\[
\dimH G_{\varphi_n}(E_n)=d_n+1,
\qquad
\|\varphi_n\|_\infty=1.
\]
Tietze's extension theorem gives \(\psi_n\in C(K)\) such that
\[
\psi_n|_{E_n}=\varphi_n,\qquad
\psi_n|_{K\setminus I_n}=0,\qquad
\|\psi_n\|_\infty=1.
\]

Partition \(\NN\) into pairwise disjoint infinite sets
\(\NN=B_1\cup B_2\cup\cdots\).  For \(n\in B_j\), choose coefficients
\(c_{j,n}\in(0,1]\) such that
\[
\sup_{n\in B_j}c_{j,n}=1,
\qquad
c_{j,n}\to0 \quad(n\to\infty,\ n\in B_j).
\]
Define \(v_j\in C(K)\) by
\[
v_j(x)=
\begin{cases}
c_{j,n}\psi_n(x),& x\in K\cap I_n\text{ for some }n\in B_j,\\
0,&\text{otherwise}.
\end{cases}
\]
The only possible accumulation point of the intervals \(I_n\) is \(x_0\), and
the coefficients tend to \(0\) along each \(B_j\), so \(v_j\) is continuous on
\(K\) and \(\|v_j\|_\infty=1\).

Define
\[
T:c_0\longrightarrow C(K),\qquad
T((a_j)_{j\geq1})=\sum_{j=1}^\infty a_jv_j.
\]
The sum is pointwise locally finite away from \(x_0\).  Continuity at \(x_0\)
follows from \(a_j\to0\) and \(c_{j,n}\to0\) on each fixed block.  Because the
supports of the \(v_j\)'s are disjoint,
\[
\|T(a)\|_\infty=\sup_j |a_j|.
\]
Thus \(T\) is an isometric embedding and \(T(c_0)\) is closed in \(C(K)\).

If \(a=(a_j)\neq0\), choose \(j_0\) with \(a_{j_0}\neq0\).  For every
\(n\in B_{j_0}\), on \(E_n\) we have
\[
T(a)=a_{j_0}c_{j_0,n}\varphi_n.
\]
Nonzero scalar multiplication does not change Hausdorff dimension, hence
\[
\dimH G_{T(a)}(E_n)=d_n+1,\qquad n\in B_{j_0}.
\]
Since every infinite block \(B_{j_0}\) contains indices tending to infinity,
\[
\dimH G_{T(a)}(K)\geq \sup_{n\in B_{j_0}}(d_n+1)=d+1.
\]
The reverse inequality is (F2), so \(\dimH G_{T(a)}(K)=d+1\).

Finally, extend functions linearly over every bounded component of
\([0,1]\setminus K\) and constantly over endpoint components.  This canonical
operator \(\cE_K:C(K)\to C[0,1]\) is a linear isometric embedding.  Put
\[
X_K=\cE_K(T(c_0)).
\]
Then \(X_K\) is closed in \(C[0,1]\), isomorphic to \(c_0\), satisfies the same
graph-dimension conclusion on \(K\), and is affine on the complementary
components.
\end{proof}

\section*{Main Theorem}

\begin{theorem}
Let \(1<s\leq2\).  Then \(H_s[0,1]\) is \((p,\cc)\)-spaceable for every finite
\(p\in\NN\).  In particular, Question 3.19 has an affirmative answer.
\end{theorem}

\begin{proof}
Let \(E\subset C[0,1]\) be a finite-dimensional subspace such that
\[
E\subset H_s[0,1]\cup\{0\}.
\]
Let \(p=\dim E\) and choose a basis \(f_1,\ldots,f_p\) of \(E\).  Applying the
imported proposition gives a compact set \(K_0\subset[0,1]\) such that
\[
\dimH K_0=s-1
\]
and, for every \(0\neq w\in E\),
\[
w|_{[0,1]\setminus K_0}\in H_s([0,1]\setminus K_0).
\]
Set \(d=s-1>0\).  Apply Lemma \ref{lem:c0-compact} to \(K_0\), and write
\[
X_0=X_{K_0}.
\]
Thus \(X_0\) is closed, \(\dim X_0=\cc\), every nonzero \(h\in X_0\) has
\[
\dimH G_h(K_0)=s,
\]
and every \(h\in X_0\) is affine on the connected components of
\([0,1]\setminus K_0\).

First, \(E\cap X_0=\{0\}\).  Indeed, if \(0\neq w\in E\cap X_0\), then the
choice of \(K_0\) gives
\[
\dimH G_w([0,1]\setminus K_0)=s.
\]
But \(w\in X_0\) is affine on each connected component of the open set
\([0,1]\setminus K_0\).  This open set is a countable union of intervals, and
the graph of an affine function on each interval has Hausdorff dimension \(1\).
Therefore
\[
\dimH G_w([0,1]\setminus K_0)\leq1,
\]
contradicting \(s>1\).

Let
\[
Y=E\oplus X_0.
\]
Since \(E\) is finite-dimensional and \(X_0\) is closed, \(Y\) is a closed
subspace of \(C[0,1]\).  Moreover \(E\subset Y\) and \(\dim Y=\cc\).

It remains to prove \(Y\subset H_s[0,1]\cup\{0\}\).  Let \(0\neq y\in Y\) and
write uniquely
\[
y=w+h,\qquad w\in E,\ h\in X_0.
\]

Assume first that \(w\neq0\), and put \(U=[0,1]\setminus K_0\).  On every
connected component \(J\) of \(U\), the function \(h\) is affine, so the shear
\[
(x,t)\mapsto (x,t+h(x))
\]
is bi-Lipschitz on \(J\times\RR\).  Hence
\[
\dimH G_{w+h}(J)=\dimH G_w(J).
\]
Using countable stability over the components of \(U\),
\[
\dimH G_y(U)=\dimH G_w(U)=s.
\]
On \(K_0\), (F2) gives
\[
\dimH G_y(K_0)\leq\dimH K_0+1=s.
\]
Thus
\[
\dimH G_y([0,1])=\max\{\dimH G_y(K_0),\dimH G_y(U)\}=s.
\]

Assume now that \(w=0\).  Then \(y=h\neq0\), and Lemma \ref{lem:c0-compact}
gives
\[
\dimH G_y(K_0)=s.
\]
The graph over \([0,1]\setminus K_0\) is a countable union of affine graphs, so
it has dimension at most \(1\).  Since \(s>1\), the whole graph has dimension
exactly \(s\).

Therefore every nonzero element of \(Y\) belongs to \(H_s[0,1]\).  The initial
finite-dimensional subspace \(E\) was arbitrary, so \(H_s[0,1]\) is
\((p,\cc)\)-spaceable for every finite \(p\).
\end{proof}
