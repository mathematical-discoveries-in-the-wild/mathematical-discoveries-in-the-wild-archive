\section*{Source Question}

The source paper is arXiv:1706.01662, \emph{When do triple operator integrals
take value in the trace class?}.  The question appears in Remark 24 on page 38
of the arXiv PDF.

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

In the notation of the source, the question is whether
\[
\phi\in L^\infty_\sig\bigl(\lambda_B;
\I(L^1(\lambda_A),L^\infty(\lambda_C))\bigr)
\]
implies the existence of a probability space \((\Sigma,\nu)\) and functions
\[
a\in L^\infty(\lambda_A\times\lambda_B\times\nu),\qquad
b\in L^\infty(\lambda_B\times\lambda_C\times\nu)
\]
such that
\[
\phi(t_1,t_2,t_3)=
\int_\Sigma a(t_1,t_2,\omega)b(t_2,t_3,\omega)\,d\nu(\omega)
\]
for almost every \((t_1,t_2,t_3)\).

\section*{Proof Intuition}

For each fixed middle variable \(t_2\), the hypothesis says that the slice
operator from \(L^1(\lambda_A)\) to \(L^\infty(\lambda_C)\) is integral, i.e.
it is represented by a finite measure on a compact product of weak-star dual
balls.  The only extra issue is making these representing measures depend
measurably on \(t_2\) and using one common probability space for all slices.

The proof does exactly that.  Hahn--Banach and the duality of Bochner
\(L^1\)-spaces give a bounded weak-star measurable kernel of finite measures.
Since the compact space of dual-ball pairs is metrizable, that kernel can be
randomized over \([0,1]\).  The Radon--Nikodym density of the signed kernel
against the randomized probability kernel is absorbed into one scalar factor.
The two coordinate projections of the randomized point give the two required
bounded functions \(a\) and \(b\).

\section*{A General Form}

Throughout this section \((X,\mu)\), \((Y,\eta)\), and \((Z,\rho)\) are
separable \(\sigma\)-finite measure spaces.  Put
\[
E=L^1(X,\mu),\qquad F=L^1(Z,\rho).
\]
For \(\phi\in L^\infty(X\times Y\times Z)\), let \(u_y:E\to F^*\) denote the
slice operator
\[
\langle u_y f,g\rangle
=\int_{X\times Z}\phi(x,y,z)f(x)g(z)\,d\mu(x)d\rho(z).
\]

\begin{theorem}\label{main}
Suppose
\[
y\longmapsto u_y
\]
belongs to \(L^\infty_\sig(Y;\I(E,F^*))\).  Then there is a probability space,
which may be taken to be \(([0,1],m)\), and functions
\[
a\in L^\infty(X\times Y\times[0,1]),\qquad
b\in L^\infty(Y\times Z\times[0,1])
\]
such that
\[
\phi(x,y,z)=\int_0^1 a(x,y,s)b(y,z,s)\,dm(s)
\]
for almost every \((x,y,z)\).

Moreover, if
\[
M=\operatorname*{ess\,sup}_{y\in Y} I(u_y),
\]
then \(a,b\) can be chosen so that
\(\|a\|_\infty\|b\|_\infty\le M\).
\end{theorem}

\begin{proof}
If \(M=0\), take \(a=0\) and \(b=1\).  Assume \(M>0\).

Let
\[
K=\B_{E^*}\times \B_{F^*},
\]
where both dual balls carry their weak-star topologies.  Since \(E\) and
\(F\) are separable, \(K\) is compact metrizable.  The injective tensor product
\[
G=E\wh\otimes_\eps F
\]
embeds isometrically into \(C(K)\) by
\[
J(f\otimes g)(\xi,\zeta)=\xi(f)\zeta(g).
\]
The standard identification
\[
G^*=\I(E,F^*)
\]
is the one used in Remark 24 of the source paper; see
Diestel--Uhl~\cite[Chapter VIII, Theorems 5 and 9]{DU}.

The weak-star measurable essentially bounded field \(y\mapsto u_y\) defines a
bounded functional \(\Lambda\) on \(L^1(Y;G)\) by
\[
\Lambda(h)=\int_Y \langle u_y,h(y)\rangle\,d\eta(y).
\]
Via the isometric embedding \(J\), regard \(L^1(Y;G)\) as a closed subspace of
\(L^1(Y;C(K))\).  By Hahn--Banach, extend \(\Lambda\) to a functional
\(\widetilde\Lambda\) on \(L^1(Y;C(K))\) with
\(\|\widetilde\Lambda\|=\|\Lambda\|\le M\).  The usual duality
\[
L^1(Y;C(K))^*=L^\infty_\sig(Y;M(K))
\]
gives a weak-star measurable family \((\omega_y)_{y\in Y}\) of finite complex
regular Borel measures on \(K\), with
\[
\operatorname*{ess\,sup}_{y\in Y}\|\omega_y\|\le M,
\]
such that for every \(h\in L^1(Y;C(K))\),
\[
\widetilde\Lambda(h)=\int_Y\int_K h(y,k)\,d\omega_y(k)\,d\eta(y).
\]
In particular, for almost every \(y\) and every \(f\in E\), \(g\in F\),
\begin{equation}\label{slice-measure}
\langle u_y f,g\rangle
=\int_K \xi(f)\zeta(g)\,d\omega_y(\xi,\zeta).
\end{equation}

We now put these representing measures on one probability space.  Choose a
base point \(k_0\in K\), and define a probability kernel
\[
\pi_y=\frac{|\omega_y|}{M}
 +\Bigl(1-\frac{\|\omega_y\|}{M}\Bigr)\delta_{k_0}.
\]
Let \(\gamma_y=d\omega_y/d\pi_y\).  Then \(|\gamma_y|\le M\) for almost every
\(y\).  Since \(K\) is compact metrizable, the standard randomization lemma
for probability kernels on standard Borel spaces gives a measurable map
\[
r:Y\times[0,1]\longrightarrow K
\]
such that \(r(y,\cdot)_*m=\pi_y\) for almost every \(y\).  Write
\[
r(y,s)=(\xi_{y,s},\zeta_{y,s})
\]
and set
\[
\theta(y,s)=\gamma_y(r(y,s)).
\]
Then \(\theta\in L^\infty(Y\times[0,1])\), \(\|\theta\|_\infty\le M\), and
for every \(h\in C(K)\),
\begin{equation}\label{randomized-kernel}
\int_K h(k)\,d\omega_y(k)
=\int_0^1 \theta(y,s)h(r(y,s))\,dm(s)
\end{equation}
for almost every \(y\).

The maps \((y,s)\mapsto \xi_{y,s}\in \B_{E^*}= \B_{L^\infty(X)}\) and
\((y,s)\mapsto \zeta_{y,s}\in \B_{F^*}= \B_{L^\infty(Z)}\) are weak-star
measurable and bounded.  By the standard identification of weak-star
measurable \(L^\infty\)-valued maps with scalar \(L^\infty\)-functions, choose
\[
\alpha\in L^\infty(X\times Y\times[0,1]),\qquad
\beta\in L^\infty(Y\times Z\times[0,1])
\]
representing them, i.e.
\[
\int_X \alpha(x,y,s)f(x)\,d\mu(x)=\xi_{y,s}(f),\qquad
\int_Z \beta(y,z,s)g(z)\,d\rho(z)=\zeta_{y,s}(g)
\]
for all \(f\in E\), \(g\in F\), almost every \((y,s)\).  Put
\[
a(x,y,s)=\theta(y,s)\alpha(x,y,s),\qquad
b(y,z,s)=\beta(y,z,s).
\]
Then \(\|a\|_\infty\|b\|_\infty\le M\).

Combining \eqref{slice-measure} and \eqref{randomized-kernel}, for all
\(f\in E\) and \(g\in F\), and for almost every \(y\),
\[
\begin{aligned}
\int_{X\times Z}\phi(x,y,z)f(x)g(z)\,d\mu(x)d\rho(z)
&=\int_0^1
 \theta(y,s)\xi_{y,s}(f)\zeta_{y,s}(g)\,dm(s)\\
&=\int_{X\times Z}\int_0^1
 a(x,y,s)b(y,z,s)\,dm(s)\,f(x)g(z)\,d\mu(x)d\rho(z).
\end{aligned}
\]
Since simple tensors \(f\otimes g\) span a dense subspace of
\(L^1(X\times Z)\), this identifies the two \(L^\infty(X\times Z)\)-functions
for almost every \(y\).  Equivalently,
\[
\phi(x,y,z)=\int_0^1 a(x,y,s)b(y,z,s)\,dm(s)
\]
for almost every \((x,y,z)\), as required.
\end{proof}

\section*{Answer to Remark 24}

\begin{corollary}
The natural question in Remark 24 of~\cite{CLS} has an affirmative answer.
Whenever
\[
\phi\in L^\infty_\sig\bigl(\lambda_B;
\I(L^1(\lambda_A),L^\infty(\lambda_C))\bigr),
\]
there are a probability space \((\Sigma,\nu)\) and functions
\[
a\in L^\infty(\lambda_A\times\lambda_B\times\nu),\qquad
b\in L^\infty(\lambda_B\times\lambda_C\times\nu)
\]
such that
\[
\phi(t_1,t_2,t_3)
=\int_\Sigma a(t_1,t_2,\omega)b(t_2,t_3,\omega)\,d\nu(\omega)
\]
for almost every \((t_1,t_2,t_3)\).
\end{corollary}

\begin{proof}
Apply Theorem~\ref{main} with
\[
(X,\mu)=(\sigma(A),\lambda_A),\quad
(Y,\eta)=(\sigma(B),\lambda_B),\quad
(Z,\rho)=(\sigma(C),\lambda_C).
\]
The scalar spectral measure spaces in the source paper are separable in the
sense used there, because the normal operators act on a separable Hilbert
space.
\end{proof}
