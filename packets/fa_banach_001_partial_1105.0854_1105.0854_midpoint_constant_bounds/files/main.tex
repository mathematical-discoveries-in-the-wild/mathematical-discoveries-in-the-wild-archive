\title{Improved Bounds for G\'orak's Midpoint Constant}
\author{}
\date{June 18, 2026}

\begin{document}
\maketitle

\begin{abstract}
For a bijective $M$-bi-Lipschitz map $T$ between real Banach spaces, let
$A(M)$ be the largest possible midpoint defect divided by the distance between
its two endpoints.  G\'orak's open constant is
\[
 K=\limsup_{\eps\downarrow0}\frac{A(1+\eps)}{\eps}.
\]
We prove that the limit in this definition actually exists and that
\[
 \frac e2\le K\le e.
\]
The lower bound is obtained from an absolute-value shear on a specially
renormed copy of $\R^2$.  The upper bound follows from a new reflection-orbit
inequality for $A(M)$, together with the dyadic estimate in G\'orak's theorem.
The argument does not determine the exact value of $K$.
\end{abstract}

\section*{Packet status}

This is a strengthened partial packet for G\'orak's midpoint constant
problem.  It improves the previous active packet from
\[
        1\le K\le e^{\sqrt2}/\sqrt2
\]
to
\[
        e/2\le K\le e,
\]
and proves that the defining \(\limsup\) is an actual limit.  The exact
value of \(K\) remains open in this packet.

\section{The midpoint modulus}

For $M\ge1$, define
\begin{equation}\label{eq:def-A}
 A(M)=\sup
 \frac{
 \norm{T\left(\frac{a+b}{2}\right)-\frac{T(a)+T(b)}2}}
 {\norm{a-b}},
\end{equation}
where the supremum is over all real Banach spaces $E,F$, all bijections
$T:E\to F$ satisfying
\[
 M^{-1}\norm{x-y}\le \norm{T(x)-T(y)}\le M\norm{x-y}
 \qquad(x,y\in E),
\]
and all $a\ne b$ in $E$.  Thus, with the notation of G\'orak's problem,
\begin{equation}\label{eq:K-A}
 K_{\eps}=\frac{A(1+\eps)}{\eps}.
\end{equation}
The function $A$ is nondecreasing, and the elementary triangle inequality gives
\begin{equation}\label{eq:trivial-A}
 0\le A(M)\le \frac M2.
\end{equation}

G\'orak's Theorem 2.1 gives the following dyadic estimate
\cite{Gorak}:
\begin{equation}\label{eq:dyadic}
 A(M)\le \frac{M^{k-1}}{k}
 \qquad\bigl(k\in\{2,4,8,\ldots\}\bigr).
\end{equation}
We will use two consequences.  First, if $0<h\le 1/2$, one can choose a
dyadic $k$ such that $1/2\le hk\le1$, and hence
\begin{equation}\label{eq:local-linear}
 A(e^h)\le \frac{e^{(k-1)h}}k\le \frac e k\le 2eh.
\end{equation}
Second, when $k$ itself is dyadic and $h=1/k$,
\begin{equation}\label{eq:dyadic-subsequence}
 \frac{A(e^{1/k})}{1/k}\le e^{1-1/k}.
\end{equation}

\section{A lower bound from a renormed shear}

For $t\in\R$, put
\[
 S_t(x,y)=(x+ty,y).
\]
Fix $h>0$ and define
\[
 p_h(x,y)=\max\{eh|x|,|y|\},
 \qquad
 \norm{(x,y)}_h=\sup_{t\in\R}e^{-h|t|}p_h(S_t(x,y)).
\]

\begin{lemma}\label{lem:shear-norm}
The function $\norm{\cdot}_h$ is a norm on $\R^2$.  Moreover,
\begin{equation}\label{eq:group-growth}
 e^{-h|s|}\norm{v}_h\le \norm{S_s v}_h
 \le e^{h|s|}\norm{v}_h
 \qquad(s\in\R,\ v\in\R^2),
\end{equation}
and
\begin{equation}\label{eq:basis-norms}
 \norm{(1,0)}_h=eh,
 \qquad
 \norm{(0,1)}_h=1.
\end{equation}
\end{lemma}

\begin{proof}
For each fixed $t$, the map $v\mapsto e^{-h|t|}p_h(S_t v)$ is a norm.
Their supremum is finite because
\[
 e^{-h|t|}eh|x+ty|
 \le eh|x|+eh|t|e^{-h|t|}|y|
 \le eh|x|+|y|,
\]
while $e^{-h|t|}|y|\le|y|$.  The supremum is therefore a norm.

Using $S_tS_s=S_{t+s}$ and $|t|\ge |t+s|-|s|$, we obtain
\[
 \norm{S_s v}_h
 =\sup_{t\in\R}e^{-h|t|}p_h(S_{t+s}v)
 \le e^{h|s|}\norm{v}_h.
\]
Applying this estimate to $S_{-s}S_s v=v$ gives the lower inequality in
\eqref{eq:group-growth}.

Since $S_t(1,0)=(1,0)$, the first identity in \eqref{eq:basis-norms} is
immediate.  For the second, writing $u=h|t|$ gives
\[
 \norm{(0,1)}_h
 =\sup_{u\ge0}e^{-u}\max\{eu,1\}=1,
\]
because both $e^{-u}$ and $eu e^{-u}$ are at most $1$ for $u\ge0$.
\end{proof}

\begin{proposition}\label{prop:lower}
For every $h>0$,
\[
 A(e^h)\ge \frac{eh}{2}.
\]
Consequently,
\[
 \liminf_{\eps\downarrow0}K_{\eps}\ge \frac e2.
\]
\end{proposition}

\begin{proof}
On $(\R^2,\norm{\cdot}_h)$ define
\[
 T(x,y)=(x+|y|,y).
\]
This is a bijection, with inverse $(u,v)\mapsto(u-|v|,v)$.  Given two
points, write their difference as $(r,s)$.  There is a number
$\theta\in[-1,1]$ such that
\[
 T(x,y)-T(x',y')=S_\theta(r,s);
\]
indeed, for $s\ne0$ one may take
$\theta=(|y|-|y'|)/(y-y')$, and for $s=0$ take $\theta=0$.
Lemma \ref{lem:shear-norm} therefore shows that $T$ is
$e^h$-bi-Lipschitz.

Take $a=(0,1)$ and $b=(0,-1)$.  Their midpoint is $0$, while
\[
 \frac{T(a)+T(b)}2=(1,0).
\]
By \eqref{eq:basis-norms}, the midpoint defect is $eh$ and
$\norm{a-b}_h=2$.  Hence $A(e^h)\ge eh/2$.  Since
$\eps=e^h-1$, equation \eqref{eq:K-A} yields
\[
 K_{\eps}\ge \frac{eh}{2(e^h-1)}\longrightarrow\frac e2.
\]
\end{proof}

\section{A reflection-orbit inequality}

The next estimate is the main new ingredient.

\begin{proposition}[Reflection-orbit inequality]\label{prop:orbit}
For every $M\ge1$ and every integer $n\ge1$,
\begin{equation}\label{eq:orbit}
 A(M)\left(2n-4\sum_{j=1}^n A(M^{2j})\right)
 \le M A(M^{2n}).
\end{equation}
\end{proposition}

\begin{proof}
Fix an $M$-bi-Lipschitz bijection $T:E\to F$ and a pair of distinct
points.  Translating the source and target, we may assume that the pair is
$x,-x$, that its midpoint is $0$, and that $T(0)=0$.  Put
\[
 u=\frac{T(x)+T(-x)}2,
 \qquad r=\norm{x}.
\]
Let $\Psi:E\to E$ be reflection about $0$, so $\Psi(v)=-v$, and let
$\Psi':F\to F$ be reflection about $u$,
\[
 \Psi'(w)=2u-w.
\]
Define
\[
 B=T^{-1}\Psi'T,
 \qquad S=\Psi B.
\]
The map $B$ is an involutive $M^2$-bi-Lipschitz bijection, and it exchanges
$x$ and $-x$.  It follows that $S$ is $M^2$-bi-Lipschitz and fixes both
$x$ and $-x$.  Since $B$ and $\Psi$ are involutions,
\begin{equation}\label{eq:reversible}
 S^{-1}=\Psi S\Psi.
\end{equation}

For $j\in\mathbb Z$, set $x_j=S^j(0)$.  Equation
\eqref{eq:reversible} gives $x_{-j}=-x_j$.  Also, $S^j$ is
$M^{2j}$-bi-Lipschitz and fixes $x,-x$, so its midpoint defect at this pair
is exactly $\norm{x_j}$.  Therefore
\begin{equation}\label{eq:orbit-size}
 \norm{x_j}\le 2r A(M^{2j})
 \qquad(j\ge1).
\end{equation}

The identity $B=\Psi S$ gives $B(x_{j-1})=-x_j$.  Since
$T(Bv)=2u-T(v)$, we have
\[
 T(-x_j)=2u-T(x_{j-1}).
\]
Writing
\[
 q_j=T(x_j)+T(-x_j),
\]
we obtain
\[
 T(x_j)-T(x_{j-1})=q_j-2u.
\]
Telescoping from $j=1$ to $n$, and using $T(x_0)=T(0)=0$, gives
\begin{equation}\label{eq:telescope}
 2nu=\sum_{j=1}^nq_j-T(x_n).
\end{equation}

Apply the definition of $A(M)$ to $T$ and the pair $x_j,-x_j$.  Since
$T(0)=0$,
\begin{equation}\label{eq:q-bound}
 \norm{q_j}\le4A(M)\norm{x_j}
 \le8rA(M)A(M^{2j}),
\end{equation}
where the second inequality uses \eqref{eq:orbit-size}.  Moreover,
\begin{equation}\label{eq:terminal-bound}
 \norm{T(x_n)}\le M\norm{x_n}
 \le2MrA(M^{2n}).
\end{equation}
Taking norms in \eqref{eq:telescope} and using
\eqref{eq:q-bound}--\eqref{eq:terminal-bound}, we get
\[
 2n\norm{u}
 \le8rA(M)\sum_{j=1}^nA(M^{2j})
   +2MrA(M^{2n}).
\]
The midpoint-defect ratio of the original pair is $\norm{u}/(2r)$.
Choosing $T,x$ with this ratio arbitrarily close to $A(M)$ and then taking
the supremum yields
\[
 2nA(M)
 \le4A(M)\sum_{j=1}^nA(M^{2j})+MA(M^{2n}),
\]
which is equivalent to \eqref{eq:orbit}.
\end{proof}

\section{Existence of the asymptotic limit and the upper bound}

\begin{theorem}\label{thm:main}
The limit
\[
 L=\lim_{h\downarrow0}\frac{A(e^h)}h
\]
exists and satisfies
\[
 \frac e2\le L\le e.
\]
Consequently, the limit $\lim_{\eps\downarrow0}K_{\eps}$ exists and
\begin{equation}\label{eq:main-bound}
 \boxed{\frac e2\le K\le e}.
\end{equation}
\end{theorem}

\begin{proof}
Set
\[
 f(h)=\frac{A(e^h)}h.
\]
By \eqref{eq:local-linear}, there are constants $C>0$ and $h_0>0$ such
that
\begin{equation}\label{eq:C-linear}
 A(e^s)\le Cs\qquad(0<s<h_0).
\end{equation}
For example, one may take $C=2e$ and $h_0=1/2$.

Let
\[
 \ell=\liminf_{h\downarrow0}f(h),
\]
and choose a sequence $r_m\downarrow0$ such that $f(r_m)\to\ell$.
Fix one sufficiently small value $r=r_m$, so that
$r<h_0$ and $2Cr<1$.  For $0<h<r/4$, put
\[
 n=\left\lfloor\frac r{2h}\right\rfloor,
 \qquad \rho=2nh.
\]
Then $r-2h\le\rho\le r$.  For $1\le j\le n$, estimate
\eqref{eq:C-linear} gives
\[
 \sum_{j=1}^nA(e^{2jh})
 \le C\sum_{j=1}^n2jh
 =Chn(n+1).
\]
Thus, for all sufficiently small $h$,
\begin{equation}\label{eq:positive-denominator}
 2n-4\sum_{j=1}^nA(e^{2jh})
 \ge2n\bigl(1-2Ch(n+1)\bigr)
 \ge2n(1-2Cr)>0.
\end{equation}
Apply Proposition \ref{prop:orbit} with $M=e^h$.  Since $A$ is
nondecreasing and $\rho\le r$, equations
\eqref{eq:positive-denominator} and \eqref{eq:orbit} give
\[
 f(h)
 \le \frac{e^h r}{\rho(1-2Cr)}f(r).
\]
Letting $h\downarrow0$ with $r$ fixed, we obtain
\[
 \limsup_{h\downarrow0}f(h)\le\frac{f(r)}{1-2Cr}.
\]
Now let $r=r_m\downarrow0$.  The right side tends to $\ell$, so
\[
 \limsup_{h\downarrow0}f(h)\le\ell
 =\liminf_{h\downarrow0}f(h).
\]
Hence $f(h)$ has a limit $L$.

To bound it from above, take a dyadic integer $k\to\infty$ and set
$h=1/k$.  Equation \eqref{eq:dyadic-subsequence} gives
\[
 f(1/k)\le e^{1-1/k}\longrightarrow e.
\]
Since $f$ has a limit along all $h\downarrow0$, this proves $L\le e$.
The lower bound $L\ge e/2$ follows from Proposition \ref{prop:lower}.

Finally, writing $1+\eps=e^h$ and using \eqref{eq:K-A},
\[
 K_{\eps}
 =\frac{A(e^h)}{e^h-1}
 =f(h)\frac{h}{e^h-1}
 \longrightarrow L.
\]
Thus the limsup in the original definition is an actual limit and
\eqref{eq:main-bound} follows.
\end{proof}

\section{Status and comparison}

G\'orak recorded the bounds $1/2\le K\le3$ and asked for the exact value
\cite{Gorak}.  The supplied partial argument correctly improves these to
\[
 1\le K\le \frac{e^{\sqrt2}}{\sqrt2}=2.9085\ldots.
\]
The present argument further improves the interval to
\[
 1.3591409\ldots=\frac e2
 \le K\le
 e=2.7182818\ldots
\]
and, independently of the numerical improvement, proves that the defining
limsup is a limit.

The remaining factor-two gap is genuine in the present method.  The lower
example is closely related to the sharp $1/e$ lower bound for the rank-one
numerical index of real Banach spaces \cite{ChicaMartinMeri}.  This makes
$K=e/2$ a natural candidate, but the reflection-orbit argument alone only
yields the upper bound $e$.  No claim about the exact value is made here.
