\section*{Source and Open Problem}

This packet concerns Problem 3.10 from:

\begin{quote}\raggedright
Jerzy Kakol, Arkady Leiderman, and Artur Michalak,
\emph{A note on Banach spaces $E$ admitting a continuous map from
$C_p(X)$ onto $E_w$}, arXiv:2109.06338.
\end{quote}

The problem is on page 12 of the source paper.

\begin{center}
\includegraphics[width=\textwidth]{figures/open_problem_crop.png}
\end{center}

In the notation of the source paper, $E_w$ is the Banach space $E$ endowed
with its weak topology. The source asks:

\begin{quote}
Does there exist a separable Banach space $E$ such that $E_w$ is
homeomorphic to $C_p[0,1]$?
\end{quote}

\section*{Answer}

No. We prove a stronger statement: if $K$ is any uncountable compact
metrizable $C$-space, then $C_p(K)$ is not homeomorphic to $E_w$ for any
Banach space $E$. Since $[0,1]$ is a compact metrizable $C$-space, this
settles Problem 3.10 negatively.

\section*{Proof Intuition}

Krupski's proof that $C_p(K)$ is not homeomorphic to $C_w(L)$ for compact
metrizable $C$-spaces $K$ uses a finite-support analysis of a hypothetical
homeomorphism. The coordinate functionals in that argument are the point
masses $\delta_y$ on $C(L)$, indexed by a compact space $L$.

For a general Banach space $E$ we replace the point masses by a compact
Hilbert cube $P$ of nonzero functionals in $E^*$. Each $p\in P$ gives the
weak coordinate $e\mapsto p(e)$ on $E_w$. The finite-support machinery
assigns to each $p$ finitely many points of $K$ controlling this coordinate
through the homeomorphism. On a large strongly infinite-dimensional compact
piece, these controlling points vary continuously. Since $K$ is a $C$-space,
dimension theory forces one fiber of the support map to be non-$C$.

The inverse homeomorphism supplies finitely many weak coordinates for each
point of $K$. A Hahn--Banach separation argument then shows that every
functional in the chosen fiber must lie in one fixed finite-dimensional
linear span. A compact subset of a finite-dimensional space is a $C$-space,
contradicting the non-$C$ fiber.

\section*{Imported Topological Tool}

We use the following standard finite-support decomposition theorem from the
$C_p$ literature. Krupski states the corresponding form for weak spaces
$C_w(L)$ in \cite[Theorem 1 and Lemma 1]{KrupskiWeakPointwise}, following
Marciszewski and van Mill. The proof uses only the compact metrizable
parameter space and the continuity of the scalar coordinates; therefore the
same statement applies to a compact parameter space $P\subset E^*$ with
coordinates $e\mapsto p(e)$.

\begin{lemma}[Finite-support decomposition]\label{lem:finite-support}
Let $K$ and $P$ be compact metrizable spaces, let $Z$ be a space, and let
$\Phi:C_p(K)\to Z$ be a homeomorphism. Suppose that for each $p\in P$ there
is a continuous real-valued coordinate $\lambda_p$ on $Z$, and that
$p\mapsto \lambda_p(z)$ is continuous on $P$ for each fixed $z\in Z$.

For $k,m\in\mathbb N$, set
\[
  Z_{k,m}=\{(A,p)\in [K]^{\le k}\times P:
  \lambda_p(\Phi(O_K(A;1/m)))\subset [-1,1]\},
\]
where
\[
  O_K(A;1/m)=\{f\in C_p(K): |f(x)|<1/m \text{ for all }x\in A\}.
\]
Let $C(k,m)$ be the projection of $Z_{k,m}$ to $P$, and put
$E(1,m)=C(1,m)$ and $E(k,m)=C(k,m)\setminus C(k-1,m)$ for $k>1$.

If no coordinate $\lambda_p\circ\Phi$ is bounded on all of $C_p(K)$, then
each $E(k,m)$ is covered by countably many $G_\delta$ subsets $G$ of $P$
such that on each $G$ there are continuous maps
\[
  f_1,\ldots,f_s:G\to K
\]
with the following property. For every $p\in G$, the union of all
minimal $k$-point sets $A\subset K$ satisfying $(A,p)\in Z_{k,m}$ is exactly
\[
  \alpha(p)=\{f_1(p),\ldots,f_s(p)\}.
\]
\end{lemma}

\section*{Main Theorem}

\begin{theorem}\label{thm:main}
Let $K$ be an uncountable compact metrizable $C$-space. For every Banach
space $E$, the spaces $C_p(K)$ and $E_w$ are not homeomorphic.
\end{theorem}

\begin{proof}
If $E$ is finite-dimensional, then $E_w$ is metrizable. Since $K$ is
uncountable, $C_p(K)$ is not metrizable. Thus only the infinite-dimensional
case needs proof.

Assume toward a contradiction that there is a homeomorphism
\[
  \Phi:C_p(K)\longrightarrow E_w.
\]
Composing with a translation in $E_w$, we may suppose $\Phi(0)=0$.

\smallskip
\noindent\emph{A compact Hilbert cube of nonzero weak coordinates.}
Since $E$ is infinite-dimensional, so is $E^*$. By the basic sequence theorem
there is a normalized basic sequence $(u_n^*)_{n=0}^\infty$ in $E^*$. Put
\[
  P=\left\{\frac12 u_0^*+\sum_{n=1}^\infty 2^{-n-4}t_n u_n^*:
    (t_n)\in [0,1]^{\mathbb N}\right\}.
\]
The series converges uniformly in norm, so $P$ is norm compact and hence
$\weakstar$-compact. The coefficient representation is unique because
$(u_n^*)$ is basic; therefore $P$ is homeomorphic to the Hilbert cube.
Moreover every $p\in P$ is nonzero, since the perturbation has norm at most
$\sum_{n=1}^\infty 2^{-n-4}=1/16<1/2=\norm{\frac12 u_0^*}$.
In particular $P$ is a strongly infinite-dimensional compact metrizable
space.

For $p\in P$, let $\lambda_p:E_w\to\mathbb R$ be the weak coordinate
\[
  \lambda_p(e)=p(e).
\]
For fixed $e\in E$, the map $p\mapsto p(e)$ is $\weakstar$-continuous on
$P$. Also $\lambda_p\circ\Phi$ is not bounded on $C_p(K)$, since $\Phi$ is
onto $E$ and $p$ is a nonzero linear functional on $E$.

\smallskip
\noindent\emph{Closed finite-support relations.}
For $k,m\in\mathbb N$ define
\[
  Z_{k,m}=\{(A,p)\in [K]^{\le k}\times P:
  |p(\Phi(f))|\le 1\text{ for every }f\in O_K(A;1/m)\}.
\]
Each $Z_{k,m}$ is closed. Indeed, if $(A,p)\notin Z_{k,m}$, choose
$f\in O_K(A;1/m)$ with $|p(\Phi(f))|>1$. For $A'$ close enough to $A$ in
the Vietoris topology, $A'\subset f^{-1}((-1/m,1/m))$; and for $p'$ close
enough to $p$ in the $\weakstar$ topology, $|p'(\Phi(f))|>1$. This gives an
open neighborhood of $(A,p)$ disjoint from $Z_{k,m}$.

Let $C(k,m)$ and $E(k,m)$ be as in Lemma \ref{lem:finite-support}. Since
each $p\circ\Phi$ is continuous at $0$, the sets $C(k,m)$ cover $P$. Applying
Lemma \ref{lem:finite-support}, we obtain a countable cover of $P$ by
$G_\delta$ sets $G_j$ such that on each $G_j$ the minimal supports are
enumerated by finitely many continuous maps into $K$.

\smallskip
\noindent\emph{A strongly infinite-dimensional support fiber.}
By the standard strongly infinite-dimensional compactum lemma used in
\cite[Proposition 2]{KrupskiWeakPointwise}, if a strongly
infinite-dimensional compact metrizable space is covered by countably many
$G_\delta$ sets, then one of those sets contains a strongly
infinite-dimensional compactum. Hence for some decomposition piece $G$ there
is a strongly infinite-dimensional compactum $Q\subset G$.

Let $f_1,\ldots,f_s:G\to K$ be the continuous support maps on this piece and
define
\[
  F:Q\to K^s,\qquad F(p)=(f_1(p),\ldots,f_s(p)).
\]
Finite products of compact metrizable $C$-spaces are $C$-spaces
\cite{RohmProducts}, so $K^s$ is a $C$-space. If every fiber of $F$ were a
$C$-space, then $Q$ would be a $C$-space by the standard Haver--Fedorchuk
dimension theorem for compact metrizable spaces; this is impossible because
$Q$ is strongly infinite-dimensional. Therefore there exists
$a=(a_1,\ldots,a_s)\in K^s$ such that
\[
  Q_a=F^{-1}(a)
\]
is not a $C$-space.

\smallskip
\noindent\emph{The inverse homeomorphism forces that fiber into a
finite-dimensional span.}
Fix the integer $m$ attached to the piece $G$. For every $x\in K$, continuity
of the map
\[
  e\mapsto (\Phi^{-1}(e))(x)
\]
from $E_w$ to $\mathbb R$ at $0$ gives a finite set
$H_x^m\subset E^*$ and a number $\varepsilon_x>0$ such that
\[
  e\in E,\quad |h(e)|<\varepsilon_x\text{ for every }h\in H_x^m
  \quad\Longrightarrow\quad
  |(\Phi^{-1}(e))(x)|<1/m.       \tag{1}
\]
Set
\[
  N=\Lin\Big(\bigcup_{i=1}^s H_{a_i}^m\Big)\subset E^*.
\]
This is finite-dimensional.

We claim that $Q_a\subset N$. Let $p\in Q_a$. By construction of the support
decomposition, some minimal set $A\subset \alpha(p)=\{a_1,\ldots,a_s\}$
satisfies
\[
  |p(\Phi(g))|\le 1\quad\text{for every }g\in O_K(A;1/m).       \tag{2}
\]
Suppose $p\notin N$. Since $N$ is a finite-dimensional, hence
$\weakstar$-closed, subspace of $(E^*,\sigma(E^*,E))$, the Hahn--Banach
separation theorem gives $e\in E$ such that
\[
  p(e)=2,\qquad h(e)=0\quad\text{for all }h\in N.
\]
By (1), the function $g=\Phi^{-1}(e)$ satisfies $|g(a_i)|<1/m$ for each
$a_i\in A$, so $g\in O_K(A;1/m)$. Then (2) gives $|p(e)|=|p(\Phi(g))|\le 1$,
contradicting $p(e)=2$. Thus $p\in N$, and the claim follows.

Now $Q_a$ is a compact subset of the finite-dimensional space $N$ endowed
with the restricted weak-star topology. On finite-dimensional Hausdorff
locally convex spaces all linear Hausdorff topologies coincide, so $Q_a$ is a
finite-dimensional compact metrizable space. In particular $Q_a$ is a
$C$-space. This contradicts the choice of $Q_a$ as a non-$C$-space.

The contradiction proves the theorem.
\end{proof}

\section*{Corollary for Problem 3.10}

\begin{theorem}
There is no separable Banach space $E$ such that $E_w$ is homeomorphic to
$C_p[0,1]$.
\end{theorem}

\begin{proof}
Apply Theorem \ref{thm:main} to $K=[0,1]$, which is an uncountable compact
metrizable $C$-space.
\end{proof}
