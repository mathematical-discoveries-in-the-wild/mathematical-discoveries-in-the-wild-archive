\section*{Source and targeted question}

The source paper is Choi, Choi, Jung, and Martin, \emph{On quasi norm attaining
operators between Banach spaces}, arXiv:2004.11025. It defines
$T\in \Lin(X,Y)$ to be quasi norm attaining if
\[
  \overline{T(B_X)}\cap \|T\|S_Y\neq \emptyset,
\]
equivalently if some sequence $(x_n)\subset B_X$ satisfies
$Tx_n\to y$ and $\|y\|=\|T\|$.

Problem 7.3 of the source asks:

\begin{quote}
Let $X,Y_1,Y_2$ be Banach spaces such that $\QNA(X,Y_j)$ is dense in
$\Lin(X,Y_j)$ for $j=1,2$. Is $\QNA(X,Y_1\oplus_\infty Y_2)$ dense in
$\Lin(X,Y_1\oplus_\infty Y_2)$?
\end{quote}

\section*{Proof intuition}

For an operator into an $\ell_\infty$-sum, quasi norm attainment can fail to
patch coordinatewise because one needs a single sequence along which both
coordinates converge. There are two useful positive mechanisms. The first is
classical: if one coordinate is made strictly dominant and norm attaining, then
the whole sum is norm attaining, so the other coordinate needs no hypothesis.
The second is genuinely quasi-norm-attaining: if the dominant coordinate is
only QNA, compactness of every subordinate coordinate supplies the missing
convergent subsequence. In both cases strict dominance ensures that the limit
in the product still has the full operator norm.

\section*{Main partial result}

\begin{theorem}[Norm-attaining dominant-coordinate subcase]
\label{thm:na-dominant}
Let $X,Y_1,Y_2$ be Banach spaces and let
$T=(T_1,T_2)\in \Lin(X,Y_1\oplus_\infty Y_2)$.

Assume that for some $j\in\{1,2\}$,
\[
  \|T_j\|=\|T\|=\max\{\|T_1\|,\|T_2\|\},
\]
and that $T_j\in\overline{\NA(X,Y_j)}$. Then
\[
  T\in \overline{\QNA(X,Y_1\oplus_\infty Y_2)}.
\]
\end{theorem}

\begin{proof}
By symmetry suppose $j=1$. If $T=0$, there is nothing to prove; put
$M=\|T_1\|=\|T\|>0$. Fix $\varepsilon>0$ and choose
$0<\delta<\min\{\varepsilon/10,M/8\}$. Pick
$A_1\in\NA(X,Y_1)$ with $\|A_1-T_1\|<\delta$. Set
$\alpha=4\delta/M$ and $S_1=(1+\alpha)A_1$. Then
$S_1\in\NA(X,Y_1)$,
\[
  \|S_1-T_1\|
  \leq \|A_1-T_1\|+\alpha\|A_1\|
  <6\delta<\varepsilon,
\]
and
\[
  \|S_1\|\geq (1+\alpha)(M-\delta)>M+2\delta\geq \|T_2\|+2\delta.
\]
Let $S=(S_1,T_2)$. Then $\|S-T\|<\varepsilon$ and
$\|S\|=\|S_1\|$ by strict dominance. Since $S_1$ attains its norm, choose
$x_0\in S_X$ with $\|S_1x_0\|=\|S_1\|$. We have
\[
  \|Sx_0\|_\infty
  =\max\{\|S_1x_0\|,\|T_2x_0\|\}
  =\|S_1\|=\|S\|,
\]
so $S\in\NA(X,Y_1\oplus_\infty Y_2)\subseteq
\QNA(X,Y_1\oplus_\infty Y_2)$.
\end{proof}

\begin{theorem}[Compact-subordinate $\ell_\infty$ stability]
\label{thm:compact-subordinate}
Let $X,Y_1,Y_2$ be Banach spaces and write
$Y=Y_1\oplus_\infty Y_2$. Let $T=(T_1,T_2)\in \Lin(X,Y)$.

Assume that for some $j\in\{1,2\}$,
\[
  \|T_j\|=\|T\|=\max\{\|T_1\|,\|T_2\|\},
\]
that $T_j\in \overline{\QNA(X,Y_j)}$, and that the other coordinate
$T_k$ lies in $\overline{\K(X,Y_k)}$, where $\{j,k\}=\{1,2\}$.
Then
\[
  T\in \overline{\QNA(X,Y_1\oplus_\infty Y_2)}.
\]
\end{theorem}

\begin{proof}
By symmetry suppose $j=1$ and $k=2$. If $T=0$, the conclusion is immediate, so
put $M=\|T_1\|=\|T\|>0$. Fix $\varepsilon>0$. Choose
\[
  0<\delta<\min\{\varepsilon/10,M/8\}.
\]
Since $T_1\in \overline{\QNA(X,Y_1)}$, pick $A_1\in\QNA(X,Y_1)$ with
\[
  \|A_1-T_1\|<\delta.
\]
Set $\alpha=4\delta/M$ and $S_1=(1+\alpha)A_1$. Then $S_1\in\QNA(X,Y_1)$ and
\[
  \|S_1-T_1\|
  \leq \|A_1-T_1\|+\alpha\|A_1\|
  < \delta+\frac{4\delta}{M}(M+\delta)
  < 6\delta.
\]
Moreover,
\[
  \|S_1\|
  \geq (1+\alpha)(M-\delta)
  = M+3\delta-\frac{4\delta^2}{M}
  > M+2\delta .
\]
Since $T_2\in\overline{\K(X,Y_2)}$, choose $K_2\in\K(X,Y_2)$ with
\[
  \|K_2-T_2\|<\delta .
\]
Then
\[
  \|K_2\|\leq \|T_2\|+\delta\leq M+\delta<\|S_1\|.
\]
Define $S=(S_1,K_2)\in \Lin(X,Y_1\oplus_\infty Y_2)$. We have
\[
  \|S-T\|=\max\{\|S_1-T_1\|,\|K_2-T_2\|\}<6\delta<\varepsilon,
\]
and, by strict dominance, $\|S\|=\|S_1\|$.

It remains to show that $S$ is quasi norm attaining. Since $S_1\in\QNA(X,Y_1)$,
there is a sequence $(x_n)\subset B_X$ and a vector $u_1\in Y_1$ such that
$S_1x_n\to u_1$ and $\|u_1\|=\|S_1\|$. The sequence $(K_2x_n)$ has a
norm-convergent subsequence because $K_2$ is compact; pass to a subsequence and
write $K_2x_n\to u_2$. Then
\[
  Sx_n=(S_1x_n,K_2x_n)\longrightarrow (u_1,u_2)
\]
in $Y_1\oplus_\infty Y_2$, and
\[
  \|(u_1,u_2)\|_\infty
  =\max\{\|u_1\|,\|u_2\|\}
  =\|S_1\|
  =\|S\|,
\]
since $\|u_2\|\leq \|K_2\|<\|S_1\|$. Hence $S\in \QNA(X,Y)$.
As $\varepsilon>0$ was arbitrary, $T\in\overline{\QNA(X,Y)}$.
\end{proof}

\begin{corollary}[Finite $\ell_\infty$ sums]
Let $Y=Y_1\oplus_\infty\cdots\oplus_\infty Y_m$ and
$T=(T_1,\ldots,T_m)\in\Lin(X,Y)$. Suppose that for some $j$,
$\|T_j\|=\|T\|$, $T_j\in\overline{\QNA(X,Y_j)}$, and
$T_i\in\overline{\K(X,Y_i)}$ for every $i\neq j$. Then
$T\in\overline{\QNA(X,Y)}$.
\end{corollary}

\begin{proof}
The proof is the same. Perturb and slightly scale the $j$th coordinate to be
strictly dominant. Approximate every other coordinate by a compact operator.
The quasi-norming sequence for the dominant coordinate has a subsequence on
which all finitely many compact subordinate coordinates converge.
\end{proof}

\section*{Limitations}

This is a partial positive result, not a solution of Problem 7.3. The hard
case remains when the subordinate coordinate is merely QNA-approximable and no
compactness, norm-attainment, or other synchronizing mechanism forces
convergence along the dominant coordinate's quasi-norming sequence.

The result also does not settle Problem 7.4 from the same source. In
particular, it does not prove that the proposed Gowers-space example
$G\oplus_\infty \ell_2$ has dense quasi norm attaining endomorphisms.
