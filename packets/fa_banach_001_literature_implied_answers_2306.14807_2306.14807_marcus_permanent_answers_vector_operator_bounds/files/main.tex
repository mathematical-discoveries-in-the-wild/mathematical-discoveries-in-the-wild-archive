\section*{Source questions}
Garcia, O'Loughlin, and Yu ask on page 18, Section 10, Problem 1 of
\cite{GOY} whether, for $x_1,\ldots,x_n\in\Hh$,
\[
 \frac{1}{\sqrt{n!}}\prod_{i=1}^n\|x_i\|
 \leq \|x_1\odot\cdots\odot x_n\|.
\]
Their Problem 2 asks whether, for $A_1,\ldots,A_n\in\mathcal B(\Hh)$,
\[
 \frac{1}{\sqrt{n!}}
 \sup_{\|x\|=1}\prod_{i=1}^n\|A_i x\|
 \leq \|A_1\odot\cdots\odot A_n\|.
\]

\section*{Supporting literature}
Marcus's permanent analogue of Hadamard's theorem states that every Hermitian
matrix $G=(g_{ij})_{i,j=1}^n$ that is positive semidefinite satisfies
\begin{equation}\label{eq:marcus}
 \permanent G\geq\prod_{i=1}^n g_{ii}.
\end{equation}
This is Corollary (3) on page 494 of \cite{Marcus}. An accessible modern
supporting source is Paksoy \cite{Paksoy}: Theorem 2.4 on page 73 gives the
Gram-matrix/symmetric-tensor identity, and formula (3.15) on page 76 records
\eqref{eq:marcus}.

\section*{Identification}
\begin{proposition}
Problems 1 and 2 of \cite{GOY} have affirmative answers. The constant in
Problem 1 is sharp whenever $\dim\Hh\geq n$.
\end{proposition}

\begin{proof}
Let $u=x_1\otimes\cdots\otimes x_n$ and let
\[
 \mathcal S_n=\frac1{n!}\sum_{\pi\in S_n}\widehat\pi
\]
be the orthogonal symmetrization projection used in \cite{GOY}. For the Gram
matrix $G=(\langle x_i,x_j\rangle)$, self-adjointness and idempotence of
$\mathcal S_n$ give
\begin{align*}
 \|x_1\odot\cdots\odot x_n\|^2
 &=\langle\mathcal S_nu,\mathcal S_nu\rangle
  =\langle u,\mathcal S_nu\rangle\\
 &=\frac1{n!}\sum_{\pi\in S_n}
      \prod_{i=1}^n\langle x_i,x_{\pi(i)}\rangle
  =\frac1{n!}\permanent G.
\end{align*}
Since $G$ is Hermitian positive semidefinite, \eqref{eq:marcus} yields
\[
 \|x_1\odot\cdots\odot x_n\|^2
 \geq\frac1{n!}\prod_{i=1}^n\|x_i\|^2.
\]
Taking square roots proves Problem 1. If $x_1,\ldots,x_n$ are mutually
orthogonal and nonzero, $G$ is diagonal and equality holds, proving sharpness.

Now let $x\in\Hh$ be a unit vector. The definition of the symmetric product of
operators in \cite{GOY} gives
\[
 (A_1\odot\cdots\odot A_n)x^{\otimes n}
 =(A_1x)\odot\cdots\odot(A_nx).
\]
Because $\|x^{\otimes n}\|=1$, Problem 1 applied to
$A_1x,\ldots,A_nx$ gives
\[
 \|A_1\odot\cdots\odot A_n\|
 \geq\frac1{\sqrt{n!}}\prod_{i=1}^n\|A_i x\|.
\]
Taking the supremum over unit vectors $x$ proves Problem 2.
\end{proof}
